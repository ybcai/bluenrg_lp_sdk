59 void HAL_CrashHandler(uint32_t msp, uint32_t signature)
60 {
61  volatile uint32_t * crash_info = (volatile uint32_t *)&CrashInfoRam;
62  register uint32_t reg_content;
63  /* Init to zero the crash_info RAM locations */
64  for (reg_content=0; reg_content<NMB_OF_EXCEP_RAM_WORD; reg_content++) {
65  crash_info[reg_content] = 0;
66  }
67  /* Store Crash Signature */
68  CrashInfoRam.signature = signature;
69  /* Store SP register */
70  CrashInfoRam.SP = msp;
71  for (reg_content=2; reg_content<NMB_OF_EXCEP_RAM_WORD; reg_content++) {
72  uint32_t *ptr = ((uint32_t *)msp)+(reg_content-2);
73  if ((ptr >= ((uint32_t *) _MEMORY_RAM_BEGIN_)) &&
74  (ptr <= ((uint32_t *) _MEMORY_RAM_END_)))
75  crash_info[reg_content] = *ptr;
76  }
77  NVIC_SystemReset();
78  /* Go to infinite loop when Hard Fault exception occurs */
79  while (1)
80  {}
81 }
