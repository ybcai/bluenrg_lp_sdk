562 __STATIC_INLINE uint8_t LL_EXTI_GetTrigger(uint32_t ExtiLine)
563 {
564  uint32_t level_setup, level_edge, both_edge;
565 
566  level_setup = READ_BIT(SYSCFG->IO_DTR, ExtiLine);
567  level_edge = READ_BIT(SYSCFG->IO_IEVR, ExtiLine);
568  both_edge = READ_BIT(SYSCFG->IO_IBER, ExtiLine);
569 
570  if (level_setup) {

573  } else {
574  return LL_EXTI_TRIGGER_LOW_LEVEL;
575  }
576  } else {
577  if (both_edge) {
578  return LL_EXTI_TRIGGER_BOTH_EDGE;
579  }
580  if (level_edge) {
581  return LL_EXTI_TRIGGER_RISING_EDGE;
582  }
583  }
584  return LL_EXTI_TRIGGER_FALLING_EDGE;
