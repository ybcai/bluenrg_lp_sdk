361 struct ActionPacket
362 {
363  uint8_t StateMachineNo ; /* This parameter indicates the state machine number for this action. From 0 to 7. */
364  uint8_t ActionTag; /* The configuration of the current action.
365  * Action Tag: PLL_TRIG, TXRX, TIMER_WAKEUP, WHITENING_DISABLE, INC_CHAN, TIMESTAMP_POSITION, RELATIVE */
366  uint8_t MaxReceiveLength; /* Maximum size of received payload. Applicable only for RX actions. */
367  uint32_t WakeupTime; /* Contains the wakeup time in microsecond if it is relative.
368  * If it is absolute it must be expressed in System Time (STU). */
369  uint8_t *data; /* Pointer to the array with the data to send (header, length and data field), for TX.
370  * Pointer to the array where the data received are copied, for RX.
371  * In case of RX, the array must have the max size MAX_PACKET_LENGTH. */
372  uint32_t status; /* The Status Register with the information on the action. */
373  uint32_t timestamp_receive; /* This field contains the timestamp when a packet is received.
374  * Intended to be used in the dataRoutine() callback routine. RX only. */
375  int32_t rssi; /* The rssi of the packet was received with. RX only. */
376  BlueTransStruct trans_packet; /* This is a linked list, which is going to be used by Hardware.
377  * User does not need to do anything. */
378  ActionPacket *next_true; /* Pointer to next ActionPacket if condRoutine() returns TRUE */
379  ActionPacket *next_false; /* Pointer to next ActionPacket if condRoutine() returns FALSE */
380  uint8_t (*condRoutine)(ActionPacket*); /* User callback that decide the next ActionPacket to use.
381  * It is time critical. Routine must end within 45 us. */
382  uint8_t (*dataRoutine)(ActionPacket*, ActionPacket*); /* User callback for managing data. */
383  uint8_t trans_config; /* This is for configuring the device for TX or RX. User does not need to do anything. */
384 };
