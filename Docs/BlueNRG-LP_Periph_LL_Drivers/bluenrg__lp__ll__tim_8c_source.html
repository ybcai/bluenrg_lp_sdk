289

468  /* Disable the CC1 and CC2: Reset the CC1E and CC2E Bits */

485  tmpccmr1 |= (uint32_t)(TIM_EncoderInitStruct->IC2ActiveInput >> 8U);
486  tmpccmr1 |= (uint32_t)(TIM_EncoderInitStruct->IC2Filter >> 8U);
487  tmpccmr1 |= (uint32_t)(TIM_EncoderInitStruct->IC2Prescaler >> 8U);
488

490  tmpccer &= (uint32_t)~(TIM_CCER_CC1P | TIM_CCER_CC1NP | TIM_CCER_CC2P | TIM_CCER_CC2NP);
491  tmpccer |= (uint32_t)(TIM_EncoderInitStruct->IC1Polarity);
492  tmpccer |= (uint32_t)(TIM_EncoderInitStruct->IC2Polarity << 4U);
493  tmpccer |= (uint32_t)(TIM_CCER_CC1E | TIM_CCER_CC2E);

495  /* Set encoder mode */
496  LL_TIM_SetEncoderMode(TIMx, TIM_EncoderInitStruct->EncoderMode);
497

504  return SUCCESS;
