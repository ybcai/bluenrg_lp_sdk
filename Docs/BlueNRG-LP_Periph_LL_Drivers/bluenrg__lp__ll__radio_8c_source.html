129 static int8_t read_RSSI(void)
130 {
131  int32_t rssi_dbm;
132  uint32_t rssi0 = RRM->RSSI0_DIG_OUT;
133  uint32_t rssi1 = RRM->RSSI1_DIG_OUT;
134 
135  uint32_t rssi_int16 = ((rssi1&0xFF)<<8)|(rssi0&0xFF);
136  uint32_t reg_agc = RRM->AGC_DIG_OUT;
137 
138  if ((rssi_int16 == 0U) || (reg_agc > 0xbU))
139  {
140  rssi_dbm = 127 ;
141  }
142  else
143  {
144  rssi_dbm = (int32_t)reg_agc * 6 - RSSI_OFFSET;//127 ;
145  while (rssi_int16 > 30U)
146  {
147  rssi_dbm = rssi_dbm + 6 ;
148  rssi_int16 = (rssi_int16 >> 1) ;
149  }
150  rssi_dbm = rssi_dbm + (int32_t)(uint32_t)((417U*rssi_int16 + 18080U)>>10) ;
151  }
152  return (int8_t)rssi_dbm;
153 }

168 void RADIO_IRQHandler(void)
169 {
170  uint32_t int_value = BLUE->INTERRUPT1REG;
171 
172  /* clear interrupts */
173  BLUE->INTERRUPT1REG = int_value;
174 
175  if ( (int_value & BLUE_INTERRUPT1REG_DONE) != 0) {
176 
177  ActionPacket* next, *actionPacketBackup;
178  BlueTransStruct *p;
179  uint32_t time;
180 
181  /* Copy status in order for callback to access it. */
182  globalParameters.current_action_packet->status = int_value | \
183  (BLUE->STATUSREG & BLUE_STATUSREG_PREVTRANSMIT_Msk);
184 
185  if((globalParameters.current_action_packet->condRoutine(globalParameters.current_action_packet) ) == TRUE) {
186  next = globalParameters.current_action_packet->next_true;
187  }
188  else {
189  next = globalParameters.current_action_packet->next_false;
190  }
191  /*The radio event is started. So here a check on the next packet of the event is made*/
192  if(next == NULL_0) {
193  /* timer2 off */
194  TIMER_DISABLE_RADIO_TIMERS;
195  MODIFY_REG(blueglob->BYTE4,GLOBAL_BYTE4_ACTIVE_Msk,BLUE_IDLE_0);
196  }

203  (bluedata + next->StateMachineNo)->MAXRECEIVEDLENGTH = next->MaxReceiveLength;
204 
205  /* Packet will execute on time specified by WakeupTime */
206  if((next->ActionTag & TIMER_WAKEUP ) !=0){
207  /* timer2 off */
208  TIMER_DISABLE_TIMER12;
209 
210  /* program timer at next->wakeuptime */

217  }
218  }
219  else {
220  /* back to back */
221  }
222  }
223 
224  /* Accept the packet even with CRC Error */
225  if( ( (int_value & BLUE_INTERRUPT1REG_RCVOK) != 0) || ( (int_value & BLUE_INTERRUPT1REG_RCVCRCERR) != 0) ) {
226  /* read RSSI */
227  globalParameters.current_action_packet->rssi = read_RSSI();
228 
229  /* bluedata->config = bluedata->config ^ 0x80 ; toggle NESN bit
230  bluedata->config = bluedata->config & 0x7F ; //reset NESN bit */
231 
232  /* read time stamp */
233  globalParameters.current_action_packet->timestamp_receive = TIMER_GetAnchorPoint();
234  }
235 
236  actionPacketBackup = globalParameters.current_action_packet;
237  globalParameters.current_action_packet = next;
238  actionPacketBackup->dataRoutine(actionPacketBackup, next);
239  }

242  System clock = 64 MHz and BLE clock = 16 MHz
243  a register read is necessary to end fine
244  the clear interrupt register operation,
245  due the AHB down converter latency */

878  RRM->DTB5_DIG_ENG = 0x3B;
879  retVal = 0;
880  }

886  }
