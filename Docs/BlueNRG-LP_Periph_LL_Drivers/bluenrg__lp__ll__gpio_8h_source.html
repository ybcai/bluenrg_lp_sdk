483 __STATIC_INLINE void LL_GPIO_SetPinPull(GPIO_TypeDef *GPIOx, uint32_t Pin, uint32_t Pull)
484 {
485  /* To use the GPIO register configuration for PULL-UP PULL-DOWN and NO-PULL define USE_GPIO_PULL_REGISTER,
486  * otherwise the IO PULL will be managed from the PWR peripheral block
487  */
488 #ifdef USE_GPIO_PULL_REGISTER
489  MODIFY_REG(GPIOx->PUPDR, ((Pin * Pin) * GPIO_PUPDR_PUPD0), ((Pin * Pin) * Pull));
490 #else
491  if (GPIOx == GPIOA)
492  {
493  switch (Pull)
494  {
495  case LL_GPIO_PULL_NO:
496  LL_PWR_SetNoPullA(Pin);
497  break;
498  case LL_GPIO_PULL_UP:
499  LL_PWR_EnablePUA(Pin);
500  break;
501  case LL_GPIO_PULL_DOWN:
502  LL_PWR_EnablePDA(Pin);
503  break;
504  }
505  } else {
506  switch (Pull)
507  {
508  case LL_GPIO_PULL_NO:
509  LL_PWR_SetNoPullB(Pin);
510  break;
511  case LL_GPIO_PULL_UP:
512  LL_PWR_EnablePUB(Pin);
513  break;
514  case LL_GPIO_PULL_DOWN:
515  LL_PWR_EnablePDB(Pin);
516  break;
517  }
518  }
519 #endif
520 }

549 __STATIC_INLINE uint32_t LL_GPIO_GetPinPull(GPIO_TypeDef *GPIOx, uint32_t Pin)
550 {
551  /* To use the GPIO register configuration for PULL-UP PULL-DOWN and NO-PULL define USE_GPIO_PULL_REGISTER,
552  * otherwise the IO PULL will be managed from the PWR peripheral block
553  */
554 #ifdef USE_GPIO_PULL_REGISTER
555  return (uint32_t)(READ_BIT(GPIOx->PUPDR, ((Pin * Pin) * GPIO_PUPDR_PUPD0)) / (Pin * Pin));
556 #else
557  uint32_t pu, pd;
558 
559  if (GPIOx == GPIOA)
560  {
561  pu = LL_PWR_IsEnabledPUA(Pin);
562  pd = LL_PWR_IsEnabledPDA(Pin);
563  }
564  else
565  {
566  pu = LL_PWR_IsEnabledPUB(Pin);
567  pd = LL_PWR_IsEnabledPDB(Pin);
568  }
569 
570  if ((pu & Pin) == Pin)
571  {
572  return LL_GPIO_PULL_UP;
573  }
574 
575  if ((pd & Pin) == Pin)
576  {
577  return LL_GPIO_PULL_DOWN;
578  }
579 
580  return LL_GPIO_PULL_NO;
581 #endif
582 }
