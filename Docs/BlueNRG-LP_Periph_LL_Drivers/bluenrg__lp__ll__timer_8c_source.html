152 static void _get_calibration_data(TIMER_ContextType *context)
153 {
154  int32_t period,freq;
155  period = TIMER_GET_SLOW_PERIOD;
156 
157  while(period != TIMER_GET_SLOW_PERIOD || period == 0)
158  {
159  period = TIMER_GET_SLOW_PERIOD;
160  }
161 
162  int32_t mul1 = 0x8BCF6 ;
163  int32_t b1 = period >> 8 ;
164  int32_t b2 = period & 0xff ;
165  context->period1 = ((mul1 * b1) + ((b2 * mul1)>>8) + 16 )>>5 ;
166  context->period = period;
167 
168  int32_t mult = 0x753 ;
169  freq = TIMER_GET_SLOW_FREQEUNCY;
170 
171  while(freq != TIMER_GET_SLOW_FREQEUNCY || freq == 0)
172  {
173  freq = TIMER_GET_SLOW_FREQEUNCY;
174  }
175  int32_t a1 = freq >> 6 ;
176  int32_t a2 = a1 * mult ;
177  context->freq1 = (a2 + 128) >> 8 ;
178  context->freq = freq;
179 }

635 uint8_t TIMER_SetRadioTimerValue(uint32_t timeout, BOOL event_type, BOOL cal_req)
636 {
637  uint32_t current_time, delay, radio_init_delay, device_delay, relTimeout;
638  /*choose the 2nd init duration. Check the event_type and cal. request*/
639  if(event_type == TX)
640  {
641  if(cal_req)
642  {
643  radio_init_delay = TIMER_Context.tx_cal_delay;
644  device_delay = TIMER_Context.tx_cal_delay_st;
645  }
646  else
647  {
648  radio_init_delay = TIMER_Context.tx_no_cal_delay;
649  device_delay = TIMER_Context.tx_cal_delay_st;
650  }
651  }
652  else
653  {
654  if(cal_req)
655  {
656  radio_init_delay = TIMER_Context.rx_cal_delay;
657  device_delay = TIMER_Context.tx_cal_delay_st;
658  }
659  else
660  {
661  radio_init_delay = TIMER_Context.rx_no_cal_delay;
662  device_delay = TIMER_Context.tx_cal_delay_st;
663  }
664  }
665 
666  /* At this point, it is care of the upper layers to guarantee that the timeout represents an absolute time in the future */
667  relTimeout = timeout - (uint32_t)get_system_time_and_machine(&TIMER_Context,&current_time);
668  /*Check if the timeout is beyond the wakeup time offset. Then program either the WakeUp timer or the Timer1*/
669  if (relTimeout > device_delay + TIMER_Context.hs_startup_time + MARGIN_EXT)
670  {
671  /*The timeout is after the wakeup_time_offset, So it is ok to program the wakeup timer*/
672  delay = blue_unit_conversion(relTimeout, TIMER_Context.freq1, MULT64_THR_FREQ) \
673  - blueglob->WAKEUPINITDELAY \
674  - radio_init_delay;
675  WAKEUP->BLUE_WAKEUP_TIME = ((current_time + delay) & TIMER_MAX_VALUE);
676  TIMER_BLUE_SET_REQ_MODE(0);
677  TIMER_DISABLE_TIMER12;
678  TIMER_ENABLE_BLUE_SLEEP_REQ;
679  radio_init_delay += blueglob->WAKEUPINITDELAY;
680  }
681  else
682  {
683  delay = blue_unit_conversion(relTimeout , TIMER_Context.freq1, MULT64_THR_FREQ) \
684  - TIMER_Context.tim12_delay_mt \
685  - radio_init_delay;
686  TIMER_SET_TIMER1((current_time + delay) & TIMER_MAX_VALUE);
687  TIMER_DISABLE_WAKEUP_TIMER;
688  radio_init_delay += TIMER_Context.tim12_delay_mt;
689  }
690 
691 #if HOST_WAKEUP_FIX_ENABLE
692  WAKEUP->WAKEUP_OFFSET[0] = TIMER_Context.wakeup_time_offset;
693  blueglob->BYTE4 |= 1<<7;
694  blueglob->BYTE22 = 0xF0;
695  blueglob->BYTE23 = 0xFF;
696 #endif
697 
698  /*Basic low level check with an extra margin of machine units */
699  if ((delay + radio_init_delay) < (radio_init_delay + 5))
700  {
701  TIMER_DISABLE_RADIO_TIMERS;
702  return 1;
703  }
704  TIMER_Context.last_setup_time = blue_unit_conversion(radio_init_delay, TIMER_Context.period1, MULT64_THR_PERIOD);
705  return 0;
706 }
