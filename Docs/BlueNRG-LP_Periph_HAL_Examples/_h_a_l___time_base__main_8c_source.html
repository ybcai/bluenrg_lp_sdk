262 HAL_StatusTypeDef HAL_InitTick (uint32_t TickPriority)
263 {
264  RCC_ClkInitTypeDef clkconfig;
265  uint32_t uwPrescalerValue = 0;
266  uint32_t pFLatency;
267 
268  /*Configure the TIM1 IRQ priority */
269  HAL_NVIC_SetPriority(TIM1_IRQn, TickPriority);
270 
271  /* Enable the TIM1 global Interrupt */
272  HAL_NVIC_EnableIRQ(TIM1_IRQn);
273 
274  /* Enable TIM1 clock */
275  __HAL_RCC_TIM1_CLK_ENABLE();
276 
277  /* Get clock configuration */
278  HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
279 
280 
281  /* Compute the prescaler value to have TIM1 counter clock equal to 1MHz */
282  uwPrescalerValue = (uint32_t) ((TIM_PERIPHCLK / 1000000) - 1);
283 
284  /* Initialize TIMx peripheral as follow:
285  + Prescaler = (TIM_PERIPHCLK/1000000 - 1) to have a 1MHz TIM1 counter clock.
286  + Period = (TIM1 counter clock / TIM1 output clock) - 1. to have a (1/1000) s time base.
287  + ClockDivision = 0
288  + Counter direction = Up
289  */
290  htim1.Instance = TIM1;
291  htim1.Init.Period = (1000000 / 1000) - 1;
292  htim1.Init.Prescaler = uwPrescalerValue;
293  htim1.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
294  htim1.Init.CounterMode = TIM_COUNTERMODE_UP;
295  if(HAL_TIM_Base_Init(&htim1) != HAL_OK)
296  {
297  /* Initialization Error */
298  Error_Handler();
299  }
300 
301  /* Start the TIM time Base generation in interrupt mode */
302  if(HAL_TIM_Base_Start_IT(&htim1) != HAL_OK)
303  {
304  /* Starting Error */
305  Error_Handler();
306  }
307 
308  /* Return function status */
309  return HAL_OK;
310 }
