283 static void MX_IWDG_Init(void)
284 {
285  /* WDG clock enable */
286  __HAL_RCC_WDG_CLK_ENABLE();
287 
288  /* Force WDG peripheral reset */
289  __HAL_RCC_WDG_FORCE_RESET();
290  __HAL_RCC_WDG_RELEASE_RESET();
291 
292  /* Check if WDG Reset Release flag interrupt occurred or not */
293  while(__HAL_RCC_GET_IT(RCC_IT_WDGRSTRELRDY) == 0)
294  {
295  }
296 
297  hiwdg.Instance = IWDG;
298  hiwdg.Init.Prescaler = IWDG_PRESCALER;
299  hiwdg.Init.Window = IWDG_WINDOW;
300  hiwdg.Init.Reload = IWDG_RELOAD;
301  if (HAL_IWDG_Init(&hiwdg) != HAL_OK)
302  {
303  Error_Handler();
304  }
