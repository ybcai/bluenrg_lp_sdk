306 void HAL_TIM_IC_CaptureCallback(TIM_HandleTypeDef *htim)
307 {
308  if (htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1)
309  {
310  if(uhCaptureIndex == 0)
311  {
312  /* Get the 1st Input Capture value */
313  uwIC2Value1 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
314  uhCaptureIndex = 1;
315  }
316  else if(uhCaptureIndex == 1)
317  {
318  /* Get the 2nd Input Capture value */
319  uwIC2Value2 = HAL_TIM_ReadCapturedValue(htim, TIM_CHANNEL_1);
320 
321  /* Capture computation */
322  if (uwIC2Value2 > uwIC2Value1)
323  {
324  uwDiffCapture = (uwIC2Value2 - uwIC2Value1);
325  }
326  else if (uwIC2Value2 < uwIC2Value1)
327  {
328  /* 0xFFFF is max TIM1_CCRx value */
329  uwDiffCapture = ((0xFFFF - uwIC2Value1) + uwIC2Value2) + 1;
330  }
331  else
332  {
333  /* If capture values are equal, we have reached the limit of frequency
334  measures */
335  Error_Handler();
336  }
337 
338  /* Frequency computation: for this example TIMx (TIM1) is clocked by
339  64 MHz Clk */
340  uwFrequency = TIM_PERIPHCLK / uwDiffCapture;
341  uhCaptureIndex = 0;
342  }
343  }
344 }
