104  RCC_OscInitStruct.LSEState = RCC_LSE_ON;
105 #elif defined (RTC_CLOCK_SOURCE_LSI)
106  /* Configue LSI as RTC clock soucre */

108  RCC_OscInitStruct.LSIState = RCC_LSI_ON;

111 
112  if(HAL_RCC_OscConfig(&RCC_OscInitStruct) == HAL_OK)
113  {
114  /* Enable RTC Clock */
115  __HAL_RCC_RTC_CLK_ENABLE();
116  /* The time base should be 1ms
117  Time base = ((RTC_ASYNCH_PREDIV + 1) * (RTC_SYNCH_PREDIV + 1)) / RTC_CLOCK
118  HSE as RTC clock
119  Time base = ((99 + 1) * (9 + 1)) / 1Mhz
120  = 1ms
121  LSE as RTC clock
122  Time base = ((31 + 1) * (0 + 1)) / 32.768Khz
123  = ~1ms
124  LSI as RTC clock
125  Time base = ((31 + 1) * (0 + 1)) / 32Khz
126  = 1ms
127  */
