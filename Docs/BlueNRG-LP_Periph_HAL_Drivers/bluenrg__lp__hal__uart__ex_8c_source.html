554 static void UARTEx_SetNbDataToProcess(UART_HandleTypeDef *huart)
555 {
556  uint8_t rx_fifo_depth;
557  uint8_t tx_fifo_depth;
558  uint8_t rx_fifo_threshold;
559  uint8_t tx_fifo_threshold;
560  uint8_t numerator[] = {1U, 1U, 1U, 3U, 7U, 1U, 0U, 0U};
561  uint8_t denominator[] = {8U, 4U, 2U, 4U, 8U, 1U, 1U, 1U};
562 
563  if (huart->FifoMode == UART_FIFOMODE_DISABLE)
564  {
565  huart->NbTxDataToProcess = 1U;
566  huart->NbRxDataToProcess = 1U;
567  }
568  else
569  {
570  rx_fifo_depth = RX_FIFO_DEPTH;
571  tx_fifo_depth = TX_FIFO_DEPTH;
572  rx_fifo_threshold = (uint8_t)(READ_BIT(huart->Instance->CR3, USART_CR3_RXFTCFG) >> USART_CR3_RXFTCFG_Pos);
573  tx_fifo_threshold = (uint8_t)(READ_BIT(huart->Instance->CR3, USART_CR3_TXFTCFG) >> USART_CR3_TXFTCFG_Pos);
574  huart->NbTxDataToProcess = ((uint16_t)tx_fifo_depth * numerator[tx_fifo_threshold]) / (uint16_t)denominator[tx_fifo_threshold];
575  huart->NbRxDataToProcess = ((uint16_t)rx_fifo_depth * numerator[rx_fifo_threshold]) / (uint16_t)denominator[rx_fifo_threshold];
576  }
577 }
