170 /*Check if it is time to program the pending radio timer (large timeouts).
171 1) The radio event is before the next calibration event. Then the timer must be programmed.
172 2) The calibration timer expired but the calibration didn't start and the values not updated as expected.
173  Then the next calibration event is missleading. The radio timer must be programmed.
174 3) The radio event is after the next calibration event. Then the timer will be programmed with the latest values.
175 The check on the next calibration event is made even though the calibration is disabled (max cal. interval)
176 in order to avoid counter wrapping with timeouts far in the future.
177 */
178 static void _check_radio_activity(VTIMER_RadioHandleType * timerHandle, uint8_t *expired)
179 {
180  uint64_t nextCalibrationEvent, currentTime;
181 
182  *expired = 0;
183 
184  if (timerHandle->pending){
185  nextCalibrationEvent = calibrationData.last_calibration_time + \
186  HAL_VTIMER_Context.PeriodicCalibrationInterval;
187  ATOMIC_SECTION_BEGIN();
188  currentTime = TIMER_GetCurrentSysTime();
189 
190  if((timerHandle->expiryTime < (nextCalibrationEvent + RADIO_ACTIVITY_MARGIN)) \
191  || \
192  (currentTime > (nextCalibrationEvent + CALIB_SAFE_THR)))
193  {

758 void HAL_VTIMER_Tick(void)
759 {
760  uint8_t expired = 0;
761 
762  ATOMIC_SECTION_BEGIN();
763  if(radioTimer.active){
764  if(radioTimer.expiryTime < TIMER_GetCurrentSysTime()){
765  radioTimer.active = FALSE;
766  }
767  }
768  ATOMIC_SECTION_END();
769 
770  /* Check for expired timers */
771  while (DIFF8(HAL_VTIMER_Context.expired_count,HAL_VTIMER_Context.served_count)) {
772  VTIMER_HandleType *expiredList, *curr;
773  uint8_t to_be_served = DIFF8(HAL_VTIMER_Context.expired_count,HAL_VTIMER_Context.served_count);
774 
775  HAL_VTIMER_Context.rootNode = _check_callbacks(HAL_VTIMER_Context.rootNode, &expiredList);
776 
777  /* Call all the user callbacks */
778  curr=expiredList;
779  while (curr != NULL) {
780 
781  /* Save next pointer, in case callback start the timer again */
782  VTIMER_HandleType *next = curr->next;
783  curr->active = FALSE;
784  if (curr->callback)
785  curr->callback(curr); /* we are sure a callback is set?*/
786  curr = next;
787  }
788 
789  HAL_VTIMER_Context.rootNode = _update_user_timeout(HAL_VTIMER_Context.rootNode, &expired);
790  if (expired == 1) {
791  /* A new root timer is already expired, mimic timer expire */
792  INCREMENT_EXPIRE_COUNT;
793  }
794  HAL_VTIMER_Context.served_count += to_be_served;
795  }
796 
797  /* Check for periodic calibration */
798  if (HAL_VTIMER_Context.calibration_in_progress) {
799  if (TIMER_IsCalibrationRunning() == FALSE) {
800  /* Calibration is completed */
801  HAL_VTIMER_Context.calibration_in_progress = FALSE;
802  /* Collect calibration data */
803  TIMER_UpdateCalibrationData();
804  TIMER_GetCurrentCalibrationData(&calibrationData);
805 #if HOST_WAKEUP_FIX_ENABLE
806  if(waitCal){
807  waitCal = 0;
808  radioTimer.pending = TRUE;
809  _check_radio_activity(&radioTimer,&expired);
810  HAL_VTIMER_Context.rootNode = _update_user_timeout(HAL_VTIMER_Context.rootNode, &expired);
811  }
812 #else
813  _check_radio_activity(&radioTimer,&expired);
814 #endif
815  /* Schedule next calibration event */
816  _start_timer(&calibrationTimer, TIMER_GetCurrentSysTime() + HAL_VTIMER_Context.PeriodicCalibrationInterval);
817  }
818  }
819  /* if there is a periodic calibration, start it in advance during the active phase */
820  else{
821  if(calibrationData.periodic_calibration){
822  if( TIMER_GetCurrentSysTime() > (calibrationData.last_calibration_time + TIMER_SYSTICK_PER_FIVE_SECONDS))
823  {
824  HAL_VTIMER_StopTimer(&calibrationTimer);
825  calibration_callback(&calibrationTimer);
826  }
827  }
828  }
