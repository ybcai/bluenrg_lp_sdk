486 HAL_StatusTypeDef HAL_RCC_ClockConfig(RCC_ClkInitTypeDef *RCC_ClkInitStruct, uint32_t FLatency)
487 {
488  uint32_t tickstart;
489 
490  /* Check Null pointer */
491  if(RCC_ClkInitStruct == NULL)
492  {
493  return HAL_ERROR;
494  }
495 
496  /* Check the parameters */
497  assert_param(IS_FLASH_WAIT_STATES(FLatency));
498 
499  /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
500  must be correctly programmed according to the frequency of the FLASH clock */
501 
502  /* Setup flash wait states because according the system clock frequency */
503  if (FLatency != __HAL_FLASH_GET_WAIT_STATES())
504  {
505  /* Program the new number of wait states to the LATENCY bits in the FLASH_CONFIG register */
506  __HAL_FLASH_SET_WAIT_STATES(FLatency);
507 
508  /* Get Start Tick*/
509  tickstart = HAL_GetTick();
510 
511  /* Check that the new number of wait states is taken into account to access the Flash
512  memory by reading the FLASH_CONFIG register */
513  while (__HAL_FLASH_GET_WAIT_STATES() != FLatency)
514  {
515  if ((HAL_GetTick() - tickstart) > LATENCY_TIMEOUT_VALUE)
516  {
517  return HAL_TIMEOUT;
518  }
519  }
520  }
521 
522  /*------------------------- SYSCLK Configuration ---------------------------*/
523  assert_param(IS_RCC_SYSCLKSOURCE(RCC_ClkInitStruct->SYSCLKSource));
524  assert_param(IS_RCC_SYSCLK_DIVIDER(RCC_ClkInitStruct->SYSCLKDivider));
525 
526 
527  /* HSE is selected as System Clock Source */
528  if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSE)
529  {
530  /* Check the HSE ready flag */
531  if (LL_RCC_HSE_IsReady() == 0U)
532  {
533  return HAL_ERROR;
534  }
535  }
536  /* HSI is selected as System Clock Source */
537  else
538  {
539  /* Check the HSI ready flag */
540  if (LL_RCC_HSI_IsReady() == 0U)
541  {
542  return HAL_ERROR;
543  }
544  }
545 
546  /* Enable the RC64MPLL*/
547  __HAL_RCC_RC64MPLL_ENABLE();
548 
549  /* Get Start Tick*/
550  tickstart = HAL_GetTick();
551 
552  /* Wait till RC64MPLL is ready */
553  while (LL_RCC_RC64MPLL_IsReady() == 0)
554  {
555  if ((HAL_GetTick() - tickstart) > PLL_TIMEOUT_VALUE)
556  {
557  return HAL_TIMEOUT;
558  }
559  }
560 
561  /* Configure the RC64MPLL multiplication factor */
562  __HAL_RCC_RC64MPLL_PRESC_CONFIG(RCC_ClkInitStruct->SYSCLKDivider);
563 
564  /*---------------------------------------------------------------------------*/
565

567  switch (LL_RCC_GetRC64MPLLPrescaler())

569  case LL_RCC_RC64MPLL_DIV_1:

586  break;

588  SystemCoreClock = RC64MPLL_VALUE/64;
