95 #ifdef RTC_CLOCK_SOURCE_LSE
96  /* Configue LSE as RTC clock soucre */
97  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_LSE;
98  RCC_OscInitStruct.LSEState = RCC_LSE_ON;
99 #elif defined (RTC_CLOCK_SOURCE_LSI)
100  /* Configue LSI as RTC clock soucre */
101  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_LSI1;
102  RCC_OscInitStruct.LSIState = RCC_LSI_ON;
103 #else
104 #error Please select the RTC Clock source
105 #endif /* RTC_CLOCK_SOURCE_LSE */
106 
107  if(HAL_RCC_OscConfig(&RCC_OscInitStruct) == HAL_OK)
108  {
109  /* Enable RTC Clock */
110  __HAL_RCC_RTC_CLK_ENABLE();
111  /* The time base should be 1ms
112  Time base = ((RTC_ASYNCH_PREDIV + 1) * (RTC_SYNCH_PREDIV + 1)) / RTC_CLOCK
113  HSE as RTC clock
114  Time base = ((99 + 1) * (9 + 1)) / 1MHz
115  = 1ms
116  LSE as RTC clock
117  Time base = ((31 + 1) * (0 + 1)) / 32.768KHz
118  = ~1ms
119  LSI as RTC clock
120  Time base = ((31 + 1) * (0 + 1)) / 32KHz
121  = 1ms
122  */
123  hRTC_Handle.Instance = RTC;
124  hRTC_Handle.Init.HourFormat = RTC_HOURFORMAT_24;
125  hRTC_Handle.Init.AsynchPrediv = RTC_ASYNCH_PREDIV;
126  hRTC_Handle.Init.SynchPrediv = RTC_SYNCH_PREDIV;
127  hRTC_Handle.Init.OutPut = RTC_OUTPUT_DISABLE;
128  hRTC_Handle.Init.OutPutPolarity = RTC_OUTPUT_POLARITY_HIGH;

143  counter = 0U;
