451 void HAL_PWREx_EnterDEEPSTOPMode(void)
452 {
453 
454  /* Set DeepStop mode */
455  CLEAR_BIT(PWR->CR1, PWR_CR1_LPMS);
456 
457  /* Set SLEEPDEEP bit of Cortex System Control Register */
458  SET_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
459 
460  /* Verify if the BLE IP is awake and may require a system attention */
461  if (READ_BIT(PWR->EXTSRR, PWR_EXTSRR_RFPHASEF) == 0) {
462  /* Request Wait For Interrupt */
463  __WFI();
464  }
465 
466  /* Reset SLEEPDEEP bit of Cortex System Control Register */
467  CLEAR_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
468 }

480 void HAL_PWREx_EnterSHUTDOWNMode(void)
481 {
482  /* Set Shutdown mode */
483  SET_BIT(PWR->CR1, PWR_CR1_LPMS);
484 
485  /* Set SLEEPDEEP bit of Cortex System Control Register */
486  SET_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
487 
488 /* This option is used to ensure that store operations are completed */
489 #if defined ( __CC_ARM)
490  __force_stores();
491 #endif
492 
493  /* Verify if the BLE IP is awake and may require a system attention */
494  if (READ_BIT(PWR->EXTSRR, PWR_EXTSRR_RFPHASEF) == 0) {
495  /* Request Wait For Interrupt */
496  __WFI();
497  }
498 
499  /* Following code is executed after wake up if system didn't go to SHUTDOWN
500  * or STANDBY mode according to power policy */
501 
502  /* Reset SLEEPDEEP bit of Cortex System Control Register */
503  CLEAR_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
504 }
