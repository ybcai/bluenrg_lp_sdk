294  if (((TAG_VALUE(APP_OTA_SERVICE_ADDRESS) == OTA_SERVICE_MANAGER_TAG) && (TAG_VALUE(APP_WITH_OTA_SERVICE_ADDRESS) == OTA_IN_PROGRESS_TAG))|| /* 10 */

309 int main(void)
310 {
311  pFunction Jump_To_Application;
312  uint32_t JumpAddress, appAddress;
313  uint8_t ret;
314  WakeupSourceConfig_TypeDef wakeupIO;
315  PowerSaveLevels stopLevel;
316 
317  /* Check Service manager RAM Location to verify if a jump to Service Manager has been set from the Application */
318  OTA_Check_ServiceManager_Operation();
319 
320  /* Identifies the valid application where to jump based on the OTA application validity tags values placed on
321  reserved vector table entry: OTA_TAG_VECTOR_TABLE_ENTRY_INDEX */
322  appAddress = OTA_Check_Application_Tags_Value();
323 
324  /* Check if there is a valid application where to jump */
325  if (appAddress == APP_WITH_OTA_SERVICE_ADDRESS)
326  {
327  /* Jump to user application */
328  JumpAddress = *(__IO uint32_t*) (appAddress + 4);
329  Jump_To_Application = (pFunction) JumpAddress;
330  /* Initialize user application's Stack Pointer */
331  __set_MSP(*(__IO uint32_t*) appAddress);
332  Jump_To_Application();
333 
334  /* Infinite loop */
335  while (1)
336  {
337  }
338  }
339 
340  /* Here Ota Service Manager Application is started */
341 
342  /* System initialization function */
343  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS)
344  {
345  /* Error during system clock configuration take appropriate action */
346  while(1);
347  }
348  /* Configure IOs for pwer save modes */
349  BSP_IO_Init();
350 
351  /* Init the UART peripheral */
352  BSP_COM_Init(NULL);
353 
354  ModulesInit();
355 
356  PRINTF("\r\nBlueNRG-LP BLE OTA Service Manager (version: %s)\r\n", BLE_OTA_SERVICE_MANAGER_VERSION_STRING);
357 
358  /* Init OTA Service Manager Device */
359  ret = OTA_ServiceManager_DeviceInit();
360  if (ret != BLE_STATUS_SUCCESS) {
361  PRINTF("OTA_ServiceManager_DeviceInit()--> Failed 0x%02x\r\n", ret);
362  while(1);
363  }
364 
365  /* No Wakeup Source needed */
366  wakeupIO.IO_Mask_High_polarity = 0;
367  wakeupIO.IO_Mask_Low_polarity = 0;
368  wakeupIO.RTC_enable = 0;
369 
370  /* Erase the storage area from start page to end page */
371  OTA_Erase_Flash(APP_WITH_OTA_SERVICE_PAGE_NUMBER_START,APP_WITH_OTA_SERVICE_PAGE_NUMBER_END);
372 
373 
374  while(1)
375  {
376  ModulesTick();
377 
378  /* Application tick */
379  APP_Tick();
380 
381  /* Check if the OTA firmware upgrade session has been completed */
382  if (OTA_Tick() == 1)
383  {
384  /* Jump to the new application */
385  OTA_Jump_To_New_Application();
386  }
387  /* Power Save Request */
388  HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_NOTIMER, wakeupIO, &stopLevel);
389  }
390 
391 }
