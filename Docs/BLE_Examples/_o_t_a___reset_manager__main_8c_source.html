198 static uint8_t OTA_Check_Storage_Area(uint32_t start_address, uint32_t end_address)
199 {
200  volatile uint32_t *address;
201  uint32_t i;
202 
203  for(i=start_address;i<end_address; i = i +4)
204  {
205  address = (volatile uint32_t *) i;
206  if (*address != 0xFFFFFFFF)
207  return 1; /* do flash erase */
208  }
209 
210  return 0; /* no flash erase is required */
211 }

280 int main(void)

285  /* Identifies the valid application where to jump based on the OTA application validity tags values placed on
286  reserved vector table entry: OTA_TAG_VECTOR_TABLE_ENTRY_INDEX */
287  appAddress = OTA_Check_Application_Tags_Value();

289  if (appAddress == 0) {

298  __set_MSP(*(__IO uint32_t*) appAddress);

300

302  while (1)
303  {
304  }
305 }
