372 void RTC_WakeupInit(void)
373 {
374  LL_RTC_InitTypeDef RTC_InitStruct = {0};
375  LL_RTC_TimeTypeDef RTC_TimeStruct = {0};
376  LL_RTC_DateTypeDef RTC_DateStruct = {0};
377 
378  /* The low speed clocks to use for this example is the LSE */
379 #ifndef CONFIG_HW_LS_XTAL
380 #error "No Low Speed Crystal definition!!!"
381 #endif
382 
383  if (LL_RTC_BAK_GetRegister(RTC, LL_RTC_BKP_DR1) != RTC_BKP_DATE_TIME_UPDTATED)
384  {
385  /* Enable RTC peripheral Clocks */
386  /* Enable RTC Clock */
387  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
388 
389  /* Force RTC peripheral reset */
390  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
391  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
392 
393  /* Check if RTC Reset Release flag interrupt occurred or not */
394  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0)
395  {
396  }
397 
398  /* Disable RTC registers write protection */
399  LL_RTC_DisableWriteProtection(RTC);
400 
401  /* Enter in initialization mode */
402  if (Enter_RTC_InitMode() != RTC_ERROR_NONE)
403  {
404  /* Initialization Error */
405  LED_Blinking(LED_BLINK_ERROR);
406  }
407  RTC_InitStruct.HourFormat = LL_RTC_HOURFORMAT_24HOUR;
408  RTC_InitStruct.AsynchPrescaler = RTC_ASYNCH_PREDIV;
409  RTC_InitStruct.SynchPrescaler = RTC_SYNCH_PREDIV;
410  LL_RTC_Init(RTC, &RTC_InitStruct);
411 
412  RTC_TimeStruct.TimeFormat = LL_RTC_TIME_FORMAT_AM_OR_24;
413  RTC_TimeStruct.Hours = 0x23;
414  RTC_TimeStruct.Minutes = 0x59;
415  RTC_TimeStruct.Seconds = 0x50;
416  LL_RTC_TIME_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_TimeStruct);
417  RTC_DateStruct.WeekDay = LL_RTC_WEEKDAY_FRIDAY;
418  RTC_DateStruct.Month = LL_RTC_MONTH_DECEMBER;
419  RTC_DateStruct.Day = 0x31;
420  RTC_DateStruct.Year = 0x19;
421 
422  LL_RTC_DATE_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_DateStruct);
423 
424  /* Output disabled */
425  LL_RTC_SetAlarmOutEvent(RTC, LL_RTC_ALARMOUT_DISABLE);
426 
427  /* Output polarity */
428  LL_RTC_SetOutputPolarity(RTC, LL_RTC_OUTPUTPOLARITY_PIN_HIGH);
429 
430  /* Exit of initialization mode */
431  if (Exit_RTC_InitMode() != RTC_ERROR_NONE)
432  {
433  /* Initialization Error */
434  LED_Blinking(LED_BLINK_ERROR);
435  }
436 
437  /* Enable RTC registers write protection */
438  LL_RTC_EnableWriteProtection(RTC);
439 
440  /* Writes a data in a RTC Backup data Register1 */
441  LL_RTC_BAK_SetRegister(RTC, LL_RTC_BKP_DR1, RTC_BKP_DATE_TIME_UPDTATED);
442  }
443 
444  /* Pull UP/DOWN configuration to reduce the current leakage */
445  LL_PWR_EnablePDA(LL_PWR_PUPD_IO2|LL_PWR_PUPD_IO3|LL_PWR_PUPD_IO4|LL_PWR_PUPD_IO5|LL_PWR_PUPD_IO6|LL_PWR_PUPD_IO7|LL_PWR_PUPD_IO10|LL_PWR_PUPD_IO11|LL_PWR_PUPD_IO12|LL_PWR_PUPD_IO13|LL_PWR_PUPD_IO14|LL_PWR_PUPD_IO15);
446  LL_PWR_EnablePDB(0xffff);
447 }

449 void SetRTC_WakeupTimeout(uint32_t time)
450 {
451  /* Disable write protection */
452  LL_RTC_DisableWriteProtection(RTC);
453 
454  /* Disable Wake-up Timer */
455  LL_RTC_WAKEUP_Disable(RTC);
456 
457  /* In case of interrupt mode is used, the interrupt source must disabled */
458  LL_RTC_DisableIT_WUT(RTC);
459 
460  /* Wait till RTC WUTWF flag is set */
461  while(LL_RTC_IsActiveFlag_WUTW(RTC) == 0);
462 
463  /* Clear PWR wake up Flag */
464  LL_PWR_ClearWakeupSource(LL_PWR_EWS_INT);
465 
466  /* Clear RTC Wake Up timer Flag */
467  LL_RTC_ClearFlag_WUT(RTC);
468 
469  /* Configure the Wake-up Timer counter */
470  LL_RTC_WAKEUP_SetAutoReload(RTC, time);
471 
472  /* Configure the clock source */
473  LL_RTC_WAKEUP_SetClock(RTC, LL_RTC_WAKEUPCLOCK_CKSPRE);
474 
475  /* Configure the Interrupt in the RTC_CR register */
476  LL_RTC_EnableIT_WUT(RTC);
477 
478  /* Enable the Wake-up Timer */
479  LL_RTC_WAKEUP_Enable(RTC);
480 
481  /* Enable write protection */
482  LL_RTC_EnableWriteProtection(RTC);
483 
484  /* Configure NVIC for RTC */
485  NVIC_SetPriority(RTC_IRQn, IRQ_LOW_PRIORITY);
486  NVIC_EnableIRQ(RTC_IRQn);
487 
488  /* Clear RTC Wake Up timer Flag */
489  LL_RTC_ClearFlag_WUT(RTC);
490 }

510 int main(void)
511 {
512  uint8_t ret_val;
513  WakeupSourceConfig_TypeDef wakeupIO;
514  uint32_t wakeupSources;
515  PowerSaveLevels stopLevel;
516 
517  /* System initialization function */
518  if (SystemInit(SYSCLK_64M, BLE_SYSCLK_32M) != SUCCESS) {
519  /* Error during system clock configuration take appropriate action */
520  while(1);
521  }
522 
523  /* Init the UART peripheral */
524  BSP_COM_Init(NULL);
525 
526  /* Init LED2 */
527  BSP_LED_Init(BSP_LED2);
528 
529  /* Init BUTTON 1 */
530  BSP_PB_Init(BSP_PUSH1, BUTTON_MODE_EXTI);
531 
532  /* RTC Wakeup Peripheral Init */
533  RTC_WakeupInit();
534 
535  while(1) {
536  /* Display the updated Time and Date */
537  Show_RTC_Calendar();
538 
539  TogglePinsOutputDeepstop();
540 
541  /* POWER_SAVE_LEVEL_STOP_WITH_TIMER : wake on UART (PA8)/timeout 5 sec, (RTC)/button PUSH1 (PA10) */
542  printf("Enable Power Save Request : STOP_WITH_TIMER (RTC)\r\n");
543  /* Display time Format : hh:ms:ss */
544  printf(" Time : ");
545  printf("%.2d:%.2d:%.2d", __LL_RTC_CONVERT_BCD2BIN(LL_RTC_TIME_GetHour(RTC)),
546  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_TIME_GetMinute(RTC)),
547  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_TIME_GetSecond(RTC)));
548  printf("\n\r");
549  /* Display date Format : mm-dd-yy */
550  printf(" Date : ");
551  printf("%.2d-%.2d-%.2d", __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetMonth(RTC)),
552  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetDay(RTC)),
553  2000 + __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetYear(RTC)));
554  printf("\n\r");
555 
556  while(BSP_COM_UARTBusy());
557  wakeupIO.IO_Mask_High_polarity = WAKEUP_PA10;
558  wakeupIO.IO_Mask_Low_polarity = WAKEUP_PA8;
559  wakeupIO.RTC_enable = 1;
560  SetRTC_WakeupTimeout(WAKEUP_TIMEOUT);
561  ret_val = HAL_PWR_MNGR_Request(POWER_SAVE_LEVEL_STOP_WITH_TIMER, wakeupIO, &stopLevel);
562  if (ret_val != SUCCESS)
563  printf("Error during clock config 0x%2x\r\n", ret_val);
564  PrintNegitatedLevel(stopLevel);
565  if (stopLevel >= POWER_SAVE_LEVEL_STOP_WITH_TIMER) {
566  wakeupSources = HAL_PWR_MNGR_WakeupSource();
567  PrintWakeupSource(wakeupSources);
568  }
569  DisableRTC_WakeupTimeout();
570  }
571 }
