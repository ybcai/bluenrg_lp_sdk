569 void WaitAndCheckEndOfTransfer(void)
570 {
571  /* 1 - Wait end of transmission */
572  while (ubTransmissionComplete != 1)
573  {
574  }
575  /* Disable DMA1 Tx Channel */
576  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_1);
577  /* 2 - Wait end of reception */
578  while (ubReceptionComplete != 1)
579  {
580  }
581  /* Disable DMA1 Rx Channel */
582  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_3);
583  /* 3 - Compare Transmit data to receive data */
584  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, ubNbDataToTransmit))
585  {
586  /* Processing Error */
587  printf("Processing Error.\n\r");
588  LED_Blinking(LED_BLINK_ERROR);
589  }
590  else
591  {
592  /* Turn On Led if data are well received */
593  LED_On();
594  printf("Data are well received.\n\r");
595  }
596 }
