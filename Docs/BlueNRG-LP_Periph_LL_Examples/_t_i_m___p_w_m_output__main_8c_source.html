313 __STATIC_INLINE void Configure_TIMPWMOutput(void)
314 {
315  /*************************/
316  /* GPIO AF configuration */
317  /*************************/
318  /* Enable the peripheral clock of GPIOs */
319  LL_AHB_EnableClock(LL_AHB_PERIPH_GPIOB);
320 
321  /* GPIO TIM1_CH1 configuration */
322  LL_GPIO_SetPinMode(GPIOA, LL_GPIO_PIN_4, LL_GPIO_MODE_ALTERNATE);
323  LL_GPIO_SetPinPull(GPIOA, LL_GPIO_PIN_4, LL_GPIO_PULL_UP);
324  LL_GPIO_SetPinSpeed(GPIOA, LL_GPIO_PIN_4, LL_GPIO_SPEED_FREQ_HIGH);
325  LL_GPIO_SetAFPin_0_7(GPIOA, LL_GPIO_PIN_4, LL_GPIO_AF_4);
326 
327  /***********************************************/
328  /* Configure the NVIC to handle TIM1 interrupt */
329  /***********************************************/
330  NVIC_SetPriority(TIM1_IRQn, IRQ_HIGH_PRIORITY);
331  NVIC_EnableIRQ(TIM1_IRQn);
332 
333  /******************************/
334  /* Peripheral clocks enabling */
335  /******************************/
336  /* Enable the timer peripheral clock */
337  LL_APB0_EnableClock(LL_APB0_PERIPH_TIM1);
338 
339  /***************************/
340  /* Time base configuration */
341  /***************************/
342  /* Set counter mode */
343  /* Reset value is LL_TIM_COUNTERMODE_UP */
344  LL_TIM_SetCounterMode(TIM1, LL_TIM_COUNTERMODE_UP);
345 
346  /* Set the pre-scaler value to have TIM1 counter clock equal to 10 kHz */
347  LL_TIM_SetPrescaler(TIM1, 63);
348 
349  /* Enable TIM1_ARR register preload. Writing to or reading from the */
350  /* auto-reload register accesses the preload register. The content of the */
351  /* preload register are transferred into the shadow register at each update */
352  /* event (UEV). */
353  LL_TIM_EnableARRPreload(TIM1);
354 
355  /* Set the auto-reload value */
356  LL_TIM_SetAutoReload( TIM1, __LL_TIM_CALC_ARR(TIM_PERIPHCLK, LL_TIM_GetPrescaler(TIM1), 10000) );
357 
358  /*********************************/
359  /* Output waveform configuration */
360  /*********************************/
361  /* Set output mode */
362  /* Reset value is LL_TIM_OCMODE_FROZEN */
363  LL_TIM_OC_SetMode(TIM1, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_PWM1);
364 
365  /* Set output channel polarity */
366  /* Reset value is LL_TIM_OCPOLARITY_HIGH */
367  //LL_TIM_OC_SetPolarity(TIM1, LL_TIM_CHANNEL_CH1, LL_TIM_OCPOLARITY_HIGH);
368 
369  /* Set compare value to half of the counter period (50% duty cycle ) */
370  LL_TIM_OC_SetCompareCH1(TIM1, ( (LL_TIM_GetAutoReload(TIM1) + 1 ) / 2));
371 
372  /* Enable TIM1_CCR1 register preload. Read/Write operations access the */
373  /* preload register. TIM1_CCR1 preload value is loaded in the active */
374  /* at each update event. */
375  LL_TIM_OC_EnablePreload(TIM1, LL_TIM_CHANNEL_CH1);
376 
377  /**************************/
378  /* TIM1 interrupts set-up */
379  /**************************/
380  /* Enable the capture/compare interrupt for channel 1*/
381  LL_TIM_EnableIT_CC1(TIM1);
382 
383  /**********************************/
384  /* Start output signal generation */
385  /**********************************/
386  /* Enable output channel 1 */
387  LL_TIM_CC_EnableChannel(TIM1, LL_TIM_CHANNEL_CH1);
388 
389  /* Enable the outputs (set the MOE bit in TIMx_BDTR register). */
390  LL_TIM_EnableAllOutputs(TIM1);
391 
392  /* Enable counter */
393  LL_TIM_EnableCounter(TIM1);
394 
395  /* Force update generation */
396  LL_TIM_GenerateEvent_UPDATE(TIM1);
397 }

406 __STATIC_INLINE void Configure_DutyCycle(uint32_t D)
407 {
408  uint32_t P; /* Pulse duration */
409  uint32_t T; /* PWM signal period */
410 
411  /* PWM signal period is determined by the value of the auto-reload register */
412  T = LL_TIM_GetAutoReload(TIM1) + 1;
413 
414  /* Pulse duration is determined by the value of the compare register. */
415  /* Its value is calculated in order to match the requested duty cycle. */
416  P = (D*T)/100;
417  LL_TIM_OC_SetCompareCH1(TIM1, P);
418  printf("Changes the duty cycle of the PWM signal : %d \n\r", P);
419 }

474 void TimerCaptureCompare_Callback(void)
475 {
476  uint32_t CNT, ARR;
477  CNT = LL_TIM_GetCounter(TIM1);
478  ARR = LL_TIM_GetAutoReload(TIM1);
479 
480  if (LL_TIM_OC_GetCompareCH1(TIM1) > ARR )
481  {
482  /* If capture/compare setting is greater than autoreload, there is a counter overflow and counter restarts from 0.
483  Need to add full period to counter value (ARR+1) */
484  CNT = CNT + ARR + 1;
485  }
486  uwMeasuredDutyCycle = (CNT * 100) / ( ARR + 1 );
487 }
