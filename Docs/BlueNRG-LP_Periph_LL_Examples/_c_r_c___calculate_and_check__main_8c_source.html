333 uint32_t Calculate_CRC(uint32_t BufferSize)
334 {
335  register uint32_t data = 0;
336  register uint32_t index = 0;
337 
338  /* Compute the CRC of Data Buffer array*/
339  for (index = 0; index < (BufferSize / 4); index++)
340  {
341  data = (uint32_t)((aDataBuffer[4 * index + 3] << 24) | (aDataBuffer[4 * index + 2] << 16) | (aDataBuffer[4 * index + 1] << 8) | aDataBuffer[4 * index]);
342  LL_CRC_FeedData32(CRC, data);
343  }
344 
345  /* Last bytes specific handling */
346  if ((BUFFER_SIZE % 4) != 0)
347  {
348  if (BUFFER_SIZE % 4 == 1)
349  {
350  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index]);
351  }
352  if (BUFFER_SIZE % 4 == 2)
353  {
354  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
355  }
356  if (BUFFER_SIZE % 4 == 3)
357  {
358  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
359  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index + 2]);
360  }
361  }
362  /* Return computed CRC value */
363  return(LL_CRC_ReadData32(CRC));
364 }
