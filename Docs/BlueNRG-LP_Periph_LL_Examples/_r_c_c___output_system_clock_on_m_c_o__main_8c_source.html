346 void UserButton_Callback(void)
347 {
348  register uint32_t source = 0, prescaler = 0;
349 
350  /* Get the MCO config to apply */
351  source = aMCO_Config[bMCOIndex].Source;
352  prescaler = aMCO_Config[bMCOIndex].Prescaler;
353 
354  /* Toggle LED2 to indicate a button press*/
355  LL_GPIO_TogglePin(LED2_GPIO_PORT, LED2_PIN);
356 
357  /* Update User push-button (PUSH1) variable : to be checked in waiting loop in main program */
358  ubButtonPress = 1;
359 
360  /* Select MCO clock source and prescaler */
361  LL_RCC_ConfigMCO(source, prescaler);
362 
363  /* Set new MCO config Index */
364  bMCOIndex = (bMCOIndex + 1) % RCC_MCO_CONFIG_NB;
365 }
