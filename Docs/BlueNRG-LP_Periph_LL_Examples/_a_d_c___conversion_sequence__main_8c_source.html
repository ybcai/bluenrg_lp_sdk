231 int main(void)
232 {
233  /* System initialization function */
234  if (SystemInit(SYSCLK_32M, BLE_SYSCLK_NONE) != SUCCESS) {
235  /* Error during system clock configuration take appropriate action */
236  while(1);
237  }
238 
239  /* BSP Init */
240  BSP_Init();
241 
242  /* Set systick to 1ms using system clock frequency */
243  LL_Init1msTick(SystemCoreClock);
244 
245  /* Initialize all configured peripherals */
246  APP_ADC_Init();
247  APP_DMA_Init();
248 
249  /* Set DMA transfer addresses of source and destination */
250  LL_DMA_ConfigAddresses(DMA1, LL_DMA_CHANNEL_1,
251  LL_ADC_GetOutputDataRegDS(ADC),
252  (uint32_t)&aDST_Buffer,
253  LL_DMA_DIRECTION_PERIPH_TO_MEMORY);
254 
255  /* Set DMA transfer size */
256  LL_DMA_SetDataLength(DMA1, LL_DMA_CHANNEL_1, BUFFER_SIZE);
257 
258  /* Enable DMA transfer complete/error interrupts */
259  LL_DMA_EnableIT_TC(DMA1, LL_DMA_CHANNEL_1);
260 
261  /* Start the DMA transfer */
262  LL_ADC_DMAModeDSEnable(ADC);
263  LL_DMA_EnableChannel(DMA1, LL_DMA_CHANNEL_1);
264 
265  /* Start ADC conversion */
266  LL_ADC_StartConversion(ADC);
267 
268  /* Infinite loop */
269  while (1) {
270 
271  /* Check the ADC flag End Of Sequence of conversion */
272  if( LL_ADC_IsActiveFlag_EOS(ADC) == 1) {
273  /* Clear the ADC flag End Of Sequence of conversion */
274  LL_ADC_ClearFlag_EOS(ADC);
275  }
276 
277  /* Check the DMA flag Transfer Complete on channel 1 */
278  if( LL_DMA_IsActiveFlag_TC1(DMA1) == 1) {
279  /* Printout the output value */
280  printf("ADC pin 1 %d mV\r\n", LL_ADC_GetADCConvertedValueSingle(ADC, aDST_Buffer[0], LL_ADC_VIN_RANGE_3V6, USER_DATAWIDTH));
281  printf("ADC pin 2 %d mV\r\n", LL_ADC_GetADCConvertedValueSingle(ADC, aDST_Buffer[1], LL_ADC_VIN_RANGE_3V6, USER_DATAWIDTH));
282  printf("ADC differential %d mV\r\n", LL_ADC_GetADCConvertedValueDiff(ADC, aDST_Buffer[2], LL_ADC_VIN_RANGE_3V6, USER_DATAWIDTH));
283  printf("Battery voltage %d mV\r\n\n", LL_ADC_GetADCConvertedValueBatt(ADC, aDST_Buffer[3], USER_DATAWIDTH));
284 
285  /* Clear the DMA flag Transfer Complete on channel 1 */
286  LL_DMA_ClearFlag_TC1(DMA1);
287 
288  /* Toggle the conversion/activity LED */
289  BSP_LED_Toggle(BSP_LED1);
290 
291  /* Add 100 ms of delay between each sequence of measurement */
292  LL_mDelay(200);
293 
294  /* Restart DMA channel */
295  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_1);
296  LL_DMA_SetMemoryAddress(DMA1, LL_DMA_CHANNEL_1, (uint32_t)&aDST_Buffer);
297  LL_DMA_SetDataLength(DMA1, LL_DMA_CHANNEL_1, BUFFER_SIZE);
298  LL_DMA_EnableChannel(DMA1, LL_DMA_CHANNEL_1);
299 
300  /* Restart ADC conversion */
301  LL_ADC_StartConversion(ADC);
302  }
303 
304  /* Check the ADC flag overrun of Down Sampler */
305  if( LL_ADC_IsActiveFlag_OVRDS(ADC) == 1) {
306 
307  /* Clear the ADC flag overrun of Down Sampler */
308  LL_ADC_ClearFlag_OVRDS(ADC);
309 
310  /* Turn on the LED2 if overrun occurs */
311  BSP_LED_On(BSP_LED2);
312  }
313 
314  /* Check the DMA flag Transfer Error on channel 1 */
315  if( LL_DMA_IsActiveFlag_TE1(DMA1) == 1) {
316 
317  /* Clear the DMA flag Transfer Error on channel 1 */
318  LL_ADC_ClearFlag_OVRDS(ADC);
319 
320  /* Turn on the LED2 if transfer error occurs */
321  BSP_LED_On(BSP_LED2);
322  }
323  }
324 }

355 static void APP_ADC_Init(void)
356 {
357  /* Peripheral clock enable */
358  LL_APB1_EnableClock(LL_APB1_PERIPH_ADCDIG | LL_APB1_PERIPH_ADCANA);
359 
360  /* This function must not be called on QFN32 package */
361  LL_ADC_LDOEnable(ADC);
362 
363  /* Enable the ADC */
364  LL_ADC_Enable(ADC);
365 
366  /* Configure the sample rate */
367  LL_ADC_SetSampleRate(ADC, USER_SAMPLERATE);
368 
369  /* Configure the Down Sampler data width and ratio */
370  LL_ADC_ConfigureDSDataOutput(ADC, USER_DATAWIDTH, USER_RATIO);
371 
372  /* Configure the operation mode as ADC mode (static/low frequency signal) */
373  LL_ADC_SetADCMode(ADC, LL_ADC_OP_MODE_ADC);
374 
375  /* Set the length of the conversion sequence as 4 */
376  LL_ADC_SetSequenceLength(ADC, LL_ADC_SEQ_LEN_04);
377 
378  /* Set the 1st entry of the input sequence as VINP0 */
379  LL_ADC_SetChannelSeq0(ADC, LL_ADC_CH_VINP0_TO_SINGLE_POSITIVE_INPUT);
380  LL_ADC_SetVoltageRangeSingleVinp0(ADC, LL_ADC_VIN_RANGE_3V6);
381  LL_ADC_SetCalibPoint1Gain(ADC, LL_ADC_DEFAULT_RANGE_VALUE_3V6);
382 
383  /* Set the 2nd entry of the input sequence as VINM0 */
384  LL_ADC_SetChannelSeq1(ADC, LL_ADC_CH_VINM0_TO_SINGLE_NEGATIVE_INPUT);
385  LL_ADC_SetVoltageRangeSingleVinm0(ADC, LL_ADC_VIN_RANGE_3V6);
386 
387  /* Set the 3rd entry of the input sequence as differential (VINP0-VINM0) */
388  LL_ADC_SetChannelSeq2(ADC, LL_ADC_CH_VINP0_VINM0_TO_DIFF_INPUT);
389  LL_ADC_SetVoltageRangeDiffVinp0Vinm0(ADC, LL_ADC_VIN_RANGE_3V6);
390 
391  /* Set the 4st entry of the input sequence as battery level detector */
392  LL_ADC_SetChannelSeq3(ADC, LL_ADC_CH_BATTERY_LEVEL_DETECTOR);
393 
394  /* Configuration of ADC interruptions */
395 // LL_ADC_EnableIT(ADC, LL_ADC_IRQ_EN_OVRDS | LL_ADC_IRQ_EN_EOS);
396 
397 }
