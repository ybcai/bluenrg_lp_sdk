278 static void MX_IWDG_Init(void)
279 {
280  /* Enable APB0 indipendent Watchdog peripherals clock */
281  LL_APB0_EnableClock(LL_APB0_PERIPH_WDG);
282 
283  /* Force WDG peripheral reset */
284  LL_APB0_ForceReset(LL_APB0_PERIPH_WDG);
285  LL_APB0_ReleaseReset(LL_APB0_PERIPH_WDG);
286 
287  /* Check if WDG Reset Release flag interrupt occurred or not */
288  while(LL_RCC_IsActiveFlag_WDGRSTREL() == 0)
289  {
290  }
291 
292  LL_IWDG_Enable(IWDG);
293  LL_IWDG_EnableWriteAccess(IWDG);
294  LL_IWDG_SetPrescaler(IWDG, LL_IWDG_PRESCALER_4);
295  LL_IWDG_SetWindow(IWDG, 0xFFF);
296  LL_IWDG_SetReloadCounter(IWDG, 0xFFE);
297  while (LL_IWDG_IsReady(IWDG) != 1)
298  {
299  }
300 
301  LL_IWDG_ReloadCounter(IWDG);
302 }
