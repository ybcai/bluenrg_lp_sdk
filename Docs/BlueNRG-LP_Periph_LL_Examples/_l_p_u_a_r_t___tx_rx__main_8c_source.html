230 int main(void)
231 {
232  /* System initialization function */
233  if (SystemInit(SYSCLK_32M, BLE_SYSCLK_NONE) != SUCCESS) {
234  /* Error during system clock configuration take appropriate action */
235  while(1);
236  }
237

239  LL_Init1msTick(SystemCoreClock);

242  LED_Init();
243 
244  /* LED off */
245  LED_Off();
246 
247  /* Configure LPUART1 (LPUART IP configuration and related GPIO initialization) */
248  Configure_LPUART1();
249 
250  /* Start main program loop :
251  - Wait for any character received on LPUART RX line
252  */
253 
254  /* Send Text Information on LPUART TX to PC Com port */
255  PrintInfo();
256 
257  /* Infinite loop */
258  while (1)
259  {
260  if(newCharReceived == 1)
261  {
262  /* Check if received value is corresponding to specific one : S or s */
263  if((ubReceivedChar == 'S') || (ubReceivedChar == 's'))
264  {

267  }
268  else
269  {
270  /* Turn LED2 off */
271  LED_Off();
272  }
273 
274  /* Echo received character on TX */
275  LL_LPUART_TransmitData8(LPUART1, ubReceivedChar);
276 
277  /* Wait for TC flag to be raised for last char */
278  while (!LL_LPUART_IsActiveFlag_TC(LPUART1))
279  {
280  }
281

283  }

301 void Configure_LPUART1(void)
302 {
303  /* (1) Enable GPIO clock and configures the LPUART1 pins *******************/
304  /* (TX on PB.4, RX on PB.5) **********************/
305 
306  /* Enable the peripheral clock of GPIOB */
307  LL_AHB_EnableClock(LL_AHB_PERIPH_GPIOB);
308 
309  /* Configure TX Pin as : Alternate function, High Speed, PushPull, No-Pull */
310  LL_GPIO_SetPinMode(GPIOB, LL_GPIO_PIN_4, LL_GPIO_MODE_ALTERNATE);
311  LL_GPIO_SetAFPin_0_7(GPIOB, LL_GPIO_PIN_4, LL_GPIO_AF_0);
312  LL_GPIO_SetPinSpeed(GPIOB, LL_GPIO_PIN_4, LL_GPIO_SPEED_FREQ_HIGH);
313  LL_GPIO_SetPinOutputType(GPIOB, LL_GPIO_PIN_4, LL_GPIO_OUTPUT_PUSHPULL);
314  LL_GPIO_SetPinPull(GPIOB, LL_GPIO_PIN_4, LL_GPIO_PULL_NO);

316  /* Configure RX Pin as : Alternate function, High Speed, PushPull, No-Pull */

319  LL_GPIO_SetPinSpeed(GPIOB, LL_GPIO_PIN_5, LL_GPIO_SPEED_FREQ_HIGH);
320  LL_GPIO_SetPinOutputType(GPIOB, LL_GPIO_PIN_5, LL_GPIO_OUTPUT_PUSHPULL);
321  LL_GPIO_SetPinPull(GPIOB, LL_GPIO_PIN_5, LL_GPIO_PULL_NO);

362 void PrintInfo(void)
363 {
364  uint32_t index = 0;
365 
366  /* Send characters one per one, until last char to be sent */
367  for (index = 0; index < sizeof(aTextInfo); index++)
368  {
369  /* Wait for TXE flag to be raised */
370  while (!LL_LPUART_IsActiveFlag_TXE(LPUART1))
371  {
372  }
373  /* Write character in Transmit Data register.
374  TXE flag is cleared by writing data in TDR register */
375  LL_LPUART_TransmitData8(LPUART1, aTextInfo[index]);
376  }
377 
378  /* Wait for TC flag to be raised for last char */
379  while (!LL_LPUART_IsActiveFlag_TC(LPUART1))
380  {
381  }
382 }
