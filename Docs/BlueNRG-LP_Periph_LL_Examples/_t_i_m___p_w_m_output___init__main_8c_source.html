414 __STATIC_INLINE void Configure_DutyCycle(uint32_t D)
415 {
416  uint32_t P; /* Pulse duration */
417  uint32_t T; /* PWM signal period */
418 
419  /* PWM signal period is determined by the value of the auto-reload register */
420  T = LL_TIM_GetAutoReload(TIM1) + 1;
421 
422  /* Pulse duration is determined by the value of the compare register. */
423  /* Its value is calculated in order to match the requested duty cycle. */
424  P = (D*T)/100;
425  LL_TIM_OC_SetCompareCH1(TIM1, P);
426  printf("Changes the duty cycle of the PWM signal : %d \n\r", P);
427 }

449 void TimerCaptureCompare_Callback(void)
450 {
451  uint32_t CNT, ARR;
452  CNT = LL_TIM_GetCounter(TIM1);
453  ARR = LL_TIM_GetAutoReload(TIM1);
454 
455  if (LL_TIM_OC_GetCompareCH1(TIM1) > ARR )
456  {
457  /* If capture/compare setting is greater than autoreload, there is a counter overflow and counter restarts from 0.
458  Need to add full period to counter value (ARR+1) */
459  CNT = CNT + ARR + 1;
460  }
461  uwMeasuredDutyCycle = (CNT * 100) / ( ARR + 1 );
462 }
