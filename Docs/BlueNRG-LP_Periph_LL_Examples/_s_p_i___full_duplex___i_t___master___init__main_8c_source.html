500  }
501

505  }

553 void SPI_MASTER_Rx_Callback(void)
554 {
555  /* Read character in Data register.
556  RXNE flag is cleared by reading data in DR register */
557 #if defined( CONFIG_DATASIZE_16BIT )
558  aRxBuffer[ubReceiveIndex++] = LL_SPI_ReceiveData16(SPI_MASTER);
559 #elif defined( CONFIG_DATASIZE_8BIT )
560  aRxBuffer[ubReceiveIndex++] = LL_SPI_ReceiveData8(SPI_MASTER);
561 #endif
562 
563  /* Check the end of reception */
564  if(ubNbDataToReceive == ubReceiveIndex)
565  {
566  /* Disable RXNE Interrupt */
567  LL_SPI_DisableIT_RXNE(SPI1);
568  }
