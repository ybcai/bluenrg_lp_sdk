284 static void MX_GPIO_Init(void)
285 {
286  LL_GPIO_InitTypeDef GPIO_InitStruct = {0};
287  LL_EXTI_InitTypeDef EXTI_InitStruct = {0};
288 
289  /* GPIO Ports Clock Enable */
290  LL_AHB_EnableClock(LL_AHB_PERIPH_GPIOA);
291  LL_AHB_EnableClock(LL_AHB_PERIPH_GPIOB);
292  LL_APB0_EnableClock(LL_APB0_PERIPH_SYSCFG);
293 
294  /* Configure GPIO for LED */
295  LL_GPIO_SetOutputPin(LED2_GPIO_PORT, LED2_PIN);
296  GPIO_InitStruct.Pin = LED2_PIN;
297  GPIO_InitStruct.Mode = LL_GPIO_MODE_OUTPUT;
298  GPIO_InitStruct.Speed = LL_GPIO_SPEED_FREQ_LOW;
299  GPIO_InitStruct.OutputType = LL_GPIO_OUTPUT_PUSHPULL;
300  GPIO_InitStruct.Pull = LL_GPIO_PULL_NO;
301  LL_GPIO_Init(LED2_GPIO_PORT, &GPIO_InitStruct);
302 
303  /* Configure GPIO for BUTTON */
304  LL_GPIO_SetPinPull(USER_BUTTON_GPIO_PORT, USER_BUTTON_PIN, LL_GPIO_PULL_NO);
305  LL_GPIO_SetPinMode(USER_BUTTON_GPIO_PORT, USER_BUTTON_PIN, LL_GPIO_MODE_INPUT);
306 
307  /* Enable a rising trigger External line 10 Interrupt */
308  EXTI_InitStruct.Line = LL_EXTI_LINE_PA10;
309  EXTI_InitStruct.LineCommand = ENABLE;
310  EXTI_InitStruct.Type = LL_EXTI_TYPE_EDGE;
311  EXTI_InitStruct.Trigger = LL_EXTI_TRIGGER_RISING_EDGE;
312  LL_EXTI_Init(&EXTI_InitStruct);
313 
314  /* Configure NVIC for USER_BUTTON_EXTI_IRQn */
315  NVIC_SetPriority(USER_BUTTON_EXTI_IRQn, IRQ_HIGH_PRIORITY);
316  NVIC_EnableIRQ(USER_BUTTON_EXTI_IRQn);
317

328 void Configure_MPU(void)
329 {
330  /* Disable MPU */
331  LL_MPU_Disable();
332 
333  /* Configure RAM region as Region N°0, 256KB of size and R/W region */
334  LL_MPU_ConfigRegion(EXAMPLE_RAM_REGION_NUMBER, 0x00, EXAMPLE_RAM_ADDRESS_START,
335  EXAMPLE_RAM_SIZE | portMPU_REGION_READ_WRITE | LL_MPU_ACCESS_NOT_BUFFERABLE |
336  LL_MPU_ACCESS_NOT_CACHEABLE | LL_MPU_ACCESS_NOT_SHAREABLE | LL_MPU_TEX_LEVEL0 |
337  LL_MPU_INSTRUCTION_ACCESS_ENABLE);
338 
339  /* Configure FLASH region as REGION N°1, 1MB of size and R/W region */
340  LL_MPU_ConfigRegion(EXAMPLE_FLASH_REGION_NUMBER, 0x00, EXAMPLE_FLASH_ADDRESS_START,
341  EXAMPLE_FLASH_SIZE | portMPU_REGION_READ_WRITE | LL_MPU_ACCESS_NOT_BUFFERABLE |
342  LL_MPU_ACCESS_NOT_CACHEABLE | LL_MPU_ACCESS_NOT_SHAREABLE | LL_MPU_TEX_LEVEL0 |
343  LL_MPU_INSTRUCTION_ACCESS_ENABLE);
344 
345  /* Configure Peripheral region as REGION N°2, 512MB of size, R/W and Execute
346  Never region */
347  LL_MPU_ConfigRegion(EXAMPLE_PERIPH_REGION_NUMBER, 0x00, EXAMPLE_PERIPH_ADDRESS_START,
348  EXAMPLE_PERIPH_SIZE | portMPU_REGION_READ_WRITE | LL_MPU_ACCESS_NOT_BUFFERABLE |
349  LL_MPU_ACCESS_NOT_CACHEABLE | LL_MPU_ACCESS_NOT_SHAREABLE | LL_MPU_TEX_LEVEL0 |
350  LL_MPU_INSTRUCTION_ACCESS_DISABLE);
351 
352  /* Enable MPU (any access not covered by any enabled region will cause a fault) */
353  LL_MPU_Enable(LL_MPU_CTRL_PRIVILEGED_DEFAULT);
354 }
