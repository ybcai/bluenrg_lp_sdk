407 void USART_CharReception_Callback(void)
408 {

410 
411  /* Read Received character. RXNE flag is cleared by reading of RDR register */
412  received_char = LL_USART_ReceiveData8(USART1);
413 
414  /* Check if received value is corresponding to specific one : S or s */
415  if ((received_char == 'S') || (received_char == 's'))

417  /* Turn LED2 on/off : Expected character has been received */
418  LED_Toggle();
419  }
420 
421  /* Echo received character on TX */
422  LL_USART_TransmitData8(USART1, received_char);
