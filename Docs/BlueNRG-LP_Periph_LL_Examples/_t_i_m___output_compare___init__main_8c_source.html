412 __STATIC_INLINE void Configure_OCMode(uint32_t OCMode)
413 {
414  /* Disable the counter */
415  LL_TIM_DisableCounter(TIM1);
416 
417  /* Reset the counter */
418  LL_TIM_SetCounter(TIM1, 0);
419 
420  /* Reset the compare match count */
421  uwCompareMatchCount = 0;
422 
423  /* Set the output level (active v.s. inactive) according to the new OC mode */
424  switch (OCMode)
425  {
426  case LL_TIM_OCMODE_TOGGLE:
427  /* Set the output channel to its toggles on compare match*/
428  printf("Set the output channel to its toggles on compare match.\n\r");
429  LL_TIM_OC_SetMode(TIM1, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_TOGGLE);
430  break;
431 
432  case LL_TIM_OCMODE_ACTIVE:
433  /* Set the output channel to its inactive level (LOW)*/
434  printf("Set the output channel to its inactive level (LOW).\n\r");
435  LL_TIM_OC_SetMode(TIM1, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_FORCED_INACTIVE);
436  break;
437 
438  case LL_TIM_OCMODE_INACTIVE:
439  /* Set the output channel to its active level (HIGH)*/
440  printf("Set the output channel to its active level (HIGH).\n\r");
441  LL_TIM_OC_SetMode(TIM1, LL_TIM_CHANNEL_CH1, LL_TIM_OCMODE_FORCED_ACTIVE);
442  break;
443 
444  default:
445  break;
446  }
447 
448  /* Update the output channel mode */
449  LL_TIM_OC_SetMode(TIM1, LL_TIM_CHANNEL_CH1, OCMode);
450 
451  /* Re-enable the counter */
452  LL_TIM_EnableCounter(TIM1);
453 }
