618 void WaitAndCheckEndOfTransfer(void)
619 {
620  /* 1 - Wait end of transmission */
621  while (ubTransmissionComplete != 1)
622  {
623  }
624  /* Disable DMA1 Tx Channel */
625  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_3);
626  /* 2 - Wait end of reception */
627  while (ubReceptionComplete != 1)
628  {
629  }
630  /* Disable DMA Rx Channel */
631  LL_DMA_DisableChannel(DMA1, LL_DMA_CHANNEL_1);
632  /* 3 - Compare Transmit data to receive data */
633  if (Buffercmp8((uint8_t *)aTxBuffer, (uint8_t *)aRxBuffer, sizeof(aTxBuffer)))
634  {
635  /* Processing Error */
636  LED_Blinking(LED_BLINK_ERROR);
637  }
638  else
639  {
640  /* Turn On Led if data are well received */
641  printf("Data are well received.\n\r");
642  LED_On();
643  }
644 }
