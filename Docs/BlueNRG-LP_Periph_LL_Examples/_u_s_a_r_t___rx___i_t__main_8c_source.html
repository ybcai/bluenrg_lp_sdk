549 void USART_CharReception_Callback(void)
550 {
551  __IO uint32_t received_char;
552 
553  /* Read Received character. RXNE flag is cleared by reading of RDR register */
554  received_char = LL_USART_ReceiveData8(USARTx_INSTANCE);
555 
556  /* Check if received value is corresponding to specific one : S or s */
557  if ((received_char == 'S') || (received_char == 's'))
558  {
559  /* Turn LED2 on : Expected character has been received */
560  LED_On();
561  }
562 
563  /* Echo received character on TX */
564  LL_USART_TransmitData8(USARTx_INSTANCE, received_char);
565 }
