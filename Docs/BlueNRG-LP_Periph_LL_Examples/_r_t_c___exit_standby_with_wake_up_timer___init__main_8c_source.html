345 void Configure_RTC(void)

355 
356  /* Force RTC peripheral reset */
357  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
358  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
359 
360  /* Check if RTC Reset Release flag interrupt occurred or not */
361  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0)
362  {
363  }
364 
365  /* Configure RTC */
366  /* Disable RTC registers write protection */
367  LL_RTC_DisableWriteProtection(RTC);
368 
369  /* Set Initialization mode */
370  LL_RTC_EnableInitMode(RTC);
371  /* Check if the Initialization mode is set */
372  while (LL_RTC_IsActiveFlag_INIT(RTC) != 1)
373  {
374  }
375 
376  /* Set prescaler according to source clock */
377  LL_RTC_SetAsynchPrescaler(RTC, RTC_ASYNCH_PREDIV);
378  LL_RTC_SetSynchPrescaler(RTC, RTC_SYNCH_PREDIV);
379 
380  /* Disable wake up timer to modify it */
381  LL_RTC_WAKEUP_Disable(RTC);
382 
383  /* Wait until it is allow to modify wake up reload value */
384  while (LL_RTC_IsActiveFlag_WUTW(RTC) != 1)
385  {
386  }
387

389  LL_RTC_DisableInitMode(RTC);
390  /* Clear RSF flag */
