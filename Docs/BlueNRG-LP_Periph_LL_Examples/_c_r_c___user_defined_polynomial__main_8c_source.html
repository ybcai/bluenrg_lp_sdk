339 uint8_t Calculate_CRC(uint32_t BufferSize)
340 {
341  register uint32_t data = 0;
342  register uint32_t index = 0;
343 
344  /* Compute the CRC of Data Buffer array*/
345  for (index = 0; index < (BufferSize / 4); index++)
346  {
347  data = (uint32_t)((aDataBuffer[4 * index + 3] << 24) | (aDataBuffer[4 * index + 2] << 16) | (aDataBuffer[4 * index + 1] << 8) | aDataBuffer[4 * index]);
348  LL_CRC_FeedData32(CRC, data);
349  }
350 
351  /* Last bytes specific handling */
352  if ((BUFFER_SIZE % 4) != 0)
353  {
354  if (BUFFER_SIZE % 4 == 1)
355  {
356  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index]);
357  }
358  if (BUFFER_SIZE % 4 == 2)
359  {
360  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
361  }
362  if (BUFFER_SIZE % 4 == 3)
363  {
364  LL_CRC_FeedData16(CRC, (uint16_t)((aDataBuffer[4 * index + 1]<<8) | aDataBuffer[4 * index]));
365  LL_CRC_FeedData8(CRC, aDataBuffer[4 * index + 2]);
366  }
367  }
368 
369  /* Return computed CRC value */
370  return(LL_CRC_ReadData8(CRC));
371 }
