212 __STATIC_INLINE void Configure_TIMPWMOutput(void)
213 {
214  /* GPIO AF configuration */
215  /* Enable the peripheral clock of GPIOs */
216  LL_AHB_EnableClock(LL_AHB_PERIPH_GPIOA);
217 
218  /* GPIO TIM1_CH1 configuration */
219  /* PA1/AF4 ------> TIM1_CH4 */
220  LL_GPIO_SetPinMode(GPIOA, LL_GPIO_PIN_1, LL_GPIO_MODE_ALTERNATE);
221  LL_GPIO_SetPinPull(GPIOA, LL_GPIO_PIN_1, LL_GPIO_PULL_UP);
222  LL_GPIO_SetPinSpeed(GPIOA, LL_GPIO_PIN_1, LL_GPIO_SPEED_FREQ_HIGH);
223  LL_GPIO_SetAFPin_0_7(GPIOA, LL_GPIO_PIN_1, LL_GPIO_AF_4);
224 
225  /* Enable the timer peripheral clock */
226  LL_APB0_EnableClock(LL_APB0_PERIPH_TIM1);
227 
228  /* Set the pre-scaler value to have TIM1 counter clock equal to 10 Hz */
229  LL_TIM_SetPrescaler(TIM1, __LL_TIM_CALC_PSC(TIM_PERIPHCLK, 10));
230 
231  /* Set the auto-reload value to have a counter frequency of 5 Hz */
232  /* TIM1CLK = TIM_PERIPHCLK / (APB prescaler & multiplier) */
233  LL_TIM_SetAutoReload( TIM1, __LL_TIM_CALC_ARR(TIM_PERIPHCLK, LL_TIM_GetPrescaler(TIM1), 5));
234 
235  /* Set output mode */
236  /* Reset value is LL_TIM_OCMODE_FROZEN */
237  LL_TIM_OC_SetMode(TIM1, LL_TIM_CHANNEL_CH4, LL_TIM_OCMODE_PWM1);
238 
239  /* Set output channel polarity */
240  /* Reset value is LL_TIM_OCPOLARITY_HIGH */
241  LL_TIM_OC_SetPolarity(TIM1, LL_TIM_CHANNEL_CH4, LL_TIM_OCPOLARITY_HIGH);
242 
243  /* Set compare value to half of the counter period (25% duty cycle ) */
244  LL_TIM_OC_SetCompareCH4(TIM1, ((LL_TIM_GetAutoReload(TIM1) + 1 ) * (uint32_t)0.25));
245 
246  /* Repetition counter to update the timer registers only after a given number of cycles of the counter */
247  LL_TIM_SetRepetitionCounter(TIM1, 0x2); // 3 cycles = 3 - 1 = 2 = TIM1_RCR
248 
249  /* Set the one pulse mode : Counter stops counting at the next update event */
250  LL_TIM_SetOnePulseMode(TIM1, LL_TIM_ONEPULSEMODE_SINGLE);
251 
252  /* To be sure that the pulse will be repeate N times an update event must occur only */
253  /* when the repetition counter is reloaded with the content of TIMx_RCR register */
254 
255  /* Enable output channel 1 */
256  LL_TIM_CC_EnableChannel(TIM1, LL_TIM_CHANNEL_CH4);
257 
258  /* Enable the outputs (set the MOE bit in TIMx_BDTR register). */
259  LL_TIM_EnableAllOutputs(TIM1);
260 
261  /* Force update generation */
262  LL_TIM_GenerateEvent_UPDATE(TIM1);
263 
264  /* Enable counter */
265  LL_TIM_EnableCounter(TIM1);
266 }
