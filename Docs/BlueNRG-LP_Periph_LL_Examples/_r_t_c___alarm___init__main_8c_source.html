304  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0)
305  {
306  }

308  RTC_InitStruct.HourFormat = LL_RTC_HOURFORMAT_24HOUR;
309  RTC_InitStruct.AsynchPrescaler = RTC_ASYNCH_PREDIV;

311  LL_RTC_Init(RTC, &RTC_InitStruct);
