497 void Show_RTC_Calendar(void)
498 {
499  /* Note: need to convert in decimal value in using __LL_RTC_CONVERT_BCD2BIN helper macro */
500  /* Display time Format : hh:mm:ss */
501  sprintf((char *)aShowTime, "%.2d:%.2d:%.2d", __LL_RTC_CONVERT_BCD2BIN(LL_RTC_TIME_GetHour(RTC)),
502  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_TIME_GetMinute(RTC)),
503  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_TIME_GetSecond(RTC)));
504  /* Display date Format : mm-dd-yy */
505  sprintf((char *)aShowDate, "%.2d-%.2d-%.2d", __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetMonth(RTC)),
506  __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetDay(RTC)),
507  2000 + __LL_RTC_CONVERT_BCD2BIN(LL_RTC_DATE_GetYear(RTC)));
508 
509  // print on the UART the time and date
510  BSP_COM_Write("\n\r", 2);
511  BSP_COM_Write("\n\r", 2);
512  BSP_COM_Write((uint8_t *)aShowTime, sizeof(aShowTime)+1);
513  BSP_COM_Write("\n\r", 2);
514  BSP_COM_Write((uint8_t *)aShowDate, sizeof(aShowTime)+1);
515 }
