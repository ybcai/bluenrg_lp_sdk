240 {
241  /* System initialization function */

250  /* Configure leds */

253  BSP_LED_Init(BSP_LED3);
254 
255  /* Initialize all configured peripherals */
256  MX_GPIO_Init();
257  MX_USART1_UART_Init();

259  /*-2- Configure UART peripheral for reception process (using LL) */
260  /* Any data received will be stored "aRxBuffer" buffer : the number max of
261  data received is RXBUFFERSIZE */
262  /* Enable RXNE and Error interrupts */
263  LL_USART_EnableIT_RXNE(USART1);
264  LL_USART_EnableIT_ERROR(USART1);
265 
266  /*-3- Start the transmission process (using LL) **/
267  /* While the UART in reception process, user can transmit data from
268  "aTxStartMessage" buffer */
269  /* Start USART transmission : Will initiate TXE interrupt after TDR register is empty */
270  LL_USART_TransmitData8(USART1, aTxStartMessage[uwTxIndex++]);
271 
272  BSP_LED_On(BSP_LED1);
273 
274  /* Enable TXE interrupt */
275  LL_USART_EnableIT_TXE(USART1);
276 
277  /*-4- Wait for the end of the transfer */
278  /* USART IRQ handler is not anymore routed to HAL_UART_IRQHandler() function
279  and is now based on LL API functions use.
280  Therefore, use of HAL IT based services is no more possible. */
281  /* Once TX transfer is completed, LED1 will turn On.
282  Then, when RX transfer is completed, LED2 will turn On. */
283  while (ubTxComplete == 0)
284  {
285  }
286 
287  while (ubRxComplete == 0)
288  {
289  }
290 
291  BSP_LED_Off(BSP_LED1);
292  BSP_LED_On(BSP_LED2);
293 
294 
295  /*-5- Send the received Buffer */
296  /* Even if use of HAL IT based services is no more possible, use of HAL Polling based services
297  (as Transmit in polling mode) is still possible. */
298  if(HAL_UART_Transmit(&huart1, (uint8_t*)aRxBuffer, RXBUFFERSIZE, 1000)!= HAL_OK)
299  {
300  /* Transfer error in transmission process */

302  }
303

306  {
307  /* Transfer error in transmission process */
308  Error_Handler();
309  }
310 
311  /* Infinite loop */
312  while (1)
313  {
314  }
