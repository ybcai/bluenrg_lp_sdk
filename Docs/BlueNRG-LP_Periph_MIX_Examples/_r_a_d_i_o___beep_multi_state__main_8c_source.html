221 void multi_state_schedular(multi_state_t * multi_state)
222 {
223  uint32_t next_wakeup_time_min = multi_state->wakeup_time[0];
224  uint8_t i = 0;
225  uint8_t index = 0;
226 
227  /* Find the minimum wakeup time among the active state machine */
228  for(i = 1;i<STATEMACHINE_COUNT;i++) {
229  if((next_wakeup_time_min > multi_state->wakeup_time[i]) && ((multi_state->activity_flag >> i)&0x01)) {
230  next_wakeup_time_min = multi_state->wakeup_time[i];
231  index = i;
232  }
233  }
234  RADIO_SetChannel(actPacket[index].StateMachineNo, channel[index], 0);
235  RADIO_MakeActionPacketPending(&actPacket[index]);
236  multi_state->wakeup_time[index] +=(period_usec[index] >> 10);
237 }

270 void beep_init(uint8_t StateMachineNo)
271 {
272  sendData[StateMachineNo][0] = 0x02;
273  sendData[StateMachineNo][1] = DATA_LEN; /* Length position is fixed */
274  sendData[StateMachineNo][2] = StateMachineNo;
275 
276  for(uint8_t i = 1; i < DATA_LEN; i++) {
277  sendData[StateMachineNo][2+i] = i;
278  }
279 
280  if(encryption[StateMachineNo]) {
281  sendData[StateMachineNo][1] += MIC_FIELD_LENGTH;
282  }
283 
284  actPacket[StateMachineNo].StateMachineNo = StateMachineNo;
285  actPacket[StateMachineNo].ActionTag = RELATIVE | TIMER_WAKEUP | TXRX | PLL_TRIG ;
286  actPacket[StateMachineNo].WakeupTime = period_usec[StateMachineNo];
287  actPacket[StateMachineNo].MaxReceiveLength = 0;/* not apllied for Tx*/
288  actPacket[StateMachineNo].data = (uint8_t *)&sendData[StateMachineNo][0];
289  actPacket[StateMachineNo].next_true = NULL; /* -> points to AP[0] */
290  actPacket[StateMachineNo].next_false = NULL; /* -> points to AP[0] */
291  actPacket[StateMachineNo].condRoutine = conditionRoutine;
292  actPacket[StateMachineNo].dataRoutine = dataRoutine;
293  multi_state.activity_flag = multi_state.activity_flag | 1<<StateMachineNo;
294  multi_state.wakeup_time[StateMachineNo] = actPacket[StateMachineNo].WakeupTime;
295  RADIO_SetChannelMap(StateMachineNo, &channel_map[0]);
296  RADIO_SetChannel(StateMachineNo, channel[StateMachineNo],0);
297  RADIO_SetTxAttributes(StateMachineNo, network_id[StateMachineNo] , 0x555555);
298  RADIO_SetTxPower(MAX_OUTPUT_RF_POWER);
299 
300  if(encryption[StateMachineNo]) {
301  RADIO_SetEncryptionCount(StateMachineNo, &count_tx[0], &count_rcv[0]);
302  RADIO_SetEncryptionAttributes(StateMachineNo, enc_iv, enc_key);
303  RADIO_SetEncryptFlags(StateMachineNo,(FunctionalState)encryption[StateMachineNo],(FunctionalState)encryption[StateMachineNo]);
304  }
305 
306  RADIO_SetReservedArea(&actPacket[StateMachineNo]);
307 }
