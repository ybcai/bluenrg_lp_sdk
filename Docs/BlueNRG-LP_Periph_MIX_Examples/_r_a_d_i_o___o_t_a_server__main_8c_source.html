301 uint8_t OTA_ConnectionCallback(ActionPacket* p, ActionPacket* next)
302 {
303  /* received a packet */
304  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
305  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
306  if(p->data[0] == HEADER_CONNECTION) {
307  ota_state_machine_g = OTA_SIZE;
308  }
309  else {
310  PRINTF(">> OTA_ConnectionCallback != %02x\r\n",p->data[0]);
311  ota_state_machine_g = OTA_CONNECTION;
312  }
313  }
314  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
315  ota_state_machine_g = OTA_CONNECTION;
316  }
317  }

407 uint8_t OTA_DataRequestCallback(ActionPacket* p, ActionPacket* next)
408 {
409  static volatile uint8_t rx_valid = 0, retry = 0;
410 
411  /* received a packet */
412  if( (p->status & BLUE_STATUSREG_PREVTRANSMIT) == 0) {
413  if((p->status & BLUE_INTERRUPT1REG_RCVOK) != 0) {
414  if(p->data[0] == HEADER_DATAREQ) {
415  if(p->data[1] == 2) {
416  rx_valid = 1;
417  seq_num = (((uint16_t)p->data[2])<<8) | ((uint16_t)p->data[3]);
418  }
419  else {
420  PRINTF(">> OTA_DataRequestCallback != %d LENGTH WRONG\r\n", p->data[1]);
421  BSP_LED_Off(BSP_LED2);
422  while(1);
423  }
424  }
425  /* It have received the ACK packet from GETDATA state */
426  else if(p->data[0] == HEADER_SENDATA) {
427  PRINTF(">> OTA_DataRequestCallback get HEADER_SENDATA\r\n");
428  }
429  else {
430  PRINTF(">> OTA_DataRequestCallback != %02x\r\n", p->data[0]);
431  while(1);
432  }
433  }
434  else if(((p->status & BLUE_INTERRUPT1REG_RCVTIMEOUT) != 0) || ((p->status & BLUE_INTERRUPT1REG_RCVCRCERR) != 0)) {
435  ota_state_machine_g = OTA_DATAREQ;
436  retry++;
437  }
438  }
439  else { /* ACK transmission completed */
440  if(rx_valid) {
441  retry = 0;
442  rx_valid = 0;
443  ota_state_machine_g = OTA_SENDATA;
444  }
445  else { /* Received a HEADER_SENDATA */
446  ota_state_machine_g = OTA_DATAREQ;
447  }
448  }
449  if(retry > MAX_RETRY) { /* create a reset function */
450  retry = 0;
451  // TODO: handle the crash of the connection, once the Ymodem is opened (send abort on ymodem channel */
452  PRINTF(">> OTA_DataRequestCallback MAX RETRY\r\n");
453  // ota_state_machine_g = OTA_CONNECTION;
454  // seq_num = 0;
455  }
456  return TRUE;
457 }

543 static volatile uint32_t debugger[3+1];
544 uint8_t OTA_Tick()
545 {
546  static uint32_t ota_pkt_idx = 0;
547  static uint32_t ota_pkt_size = 0;
548  uint8_t ret;
549 
550  /* Manage the token between the two state machines */
551  if(app_token == TOKEN_TAKEN_YMODEM)
552  return 0;
553  else
554  app_token = TOKEN_TAKEN_OTA;
555 
556  if(ota_state_machine_g == OTA_CONNECTION) {
557  PRINTF("OTA_CONNECTION\r\n");
558 
559  /* state setting */
560  last_frame = 0;
561  seq_num = 0;
562  seq_num_max = image_size / BYTE_IN_FRAME;
563  if(image_size%BYTE_IN_FRAME) {
564  seq_num_max++;
565  }
566 
567  /* frame setting */
568  tx_ota_buffer[0] = HEADER_CONNECTION;
569  tx_ota_buffer[1] = 0;
570 
571  debugger[3] = ota_state_machine_g;
572  ota_state_machine_g = OTA_ONGOING;
573  RADIO_SetBackToBackTime(150);
574 
575  BSP_LED_On(BSP_LED2);
576  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, tx_ota_buffer, rx_ota_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_ConnectionCallback);
577  if(ret != SUCCESS_0) {
578  PRINTF("ERROR %d\r\n",ret);
579  }
580  }
581 
582  else if(ota_state_machine_g == OTA_SIZE) {
583  PRINTF("OTA_SIZE\r\n");
584 
585  /* frame setting */
586  tx_ota_buffer[0] = HEADER_SIZE;
587  tx_ota_buffer[1] = 4;
588  tx_ota_buffer[2] = (uint8_t)(image_size>>24);
589  tx_ota_buffer[3] = (uint8_t)(image_size>>16);
590  tx_ota_buffer[4] = (uint8_t)(image_size>>8);
591  tx_ota_buffer[5] = (uint8_t)image_size;
592 
593  debugger[3] = ota_state_machine_g;
594  ota_state_machine_g = OTA_ONGOING;
595  RADIO_SetBackToBackTime(150);
596  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, tx_ota_buffer, rx_ota_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_SizeCallback);
597  if(ret != SUCCESS_0) {
598  PRINTF("ERROR %d\r\n",ret);
599  }
600  }
601 
602  else if(ota_state_machine_g == OTA_START) {
603  PRINTF("OTA_START\r\n");
604 
605  /* frame setting */
606  tx_ota_buffer[0] = HEADER_START;
607  tx_ota_buffer[1] = 0;
608 
609  debugger[3] = ota_state_machine_g;
610  ota_state_machine_g = OTA_ONGOING;
611  RADIO_SetBackToBackTime(300);
612  ret = HAL_RADIO_ReceivePacketWithAck(OTA_CHANNEL, RX_WAKEUP_TIME, rx_ota_buffer, tx_ota_buffer, RX_TIMEOUT_DATA, MAX_LL_PACKET_LENGTH, OTA_StartCallback);
613  if(ret != SUCCESS_0) {
614  PRINTF("ERROR %d\r\n",ret);
615  }
616  }
617 
618  else if(ota_state_machine_g == OTA_DATAREQ) {
619  PRINTF("OTA_DATAREQ %d\r\n", seq_num);
620 
621  /* Manage the token between the two state machines */
622  if(ymodem_state_machine_g == YMODEM_LOAD) {
623  app_token = TOKEN_RELEASE;
624  return 0;
625  }
626 
627  /* frame setting */
628  tx_ota_buffer[0] = HEADER_DATAREQ;
629  tx_ota_buffer[1] = 0;
630  debugger[3] = ota_state_machine_g;
631  ota_state_machine_g = OTA_ONGOING;
632  RADIO_SetBackToBackTime(300);
633  ret = HAL_RADIO_ReceivePacketWithAck(OTA_CHANNEL, RX_WAKEUP_TIME, rx_ota_buffer, tx_ota_buffer, RX_TIMEOUT_DATA, MAX_LL_PACKET_LENGTH, OTA_DataRequestCallback);
634  if(ret != SUCCESS_0) {
635  PRINTF("ERROR %d\r\n",ret);
636  }
637  }
638 
639  else if(ota_state_machine_g == OTA_SENDATA) {
640  PRINTF("OTA_SENDATA %d\r\n", seq_num);
641 
642  /* pointer of buffer */
643  ota_pkt_idx = (seq_num * BYTE_IN_FRAME);
644 
645  /* size of buffer */
646  if((ota_pkt_idx + BYTE_IN_FRAME) <= data_len) {
647  debugger[0]++;
648  ota_pkt_size = BYTE_IN_FRAME + 2; /* +2 because the sequence number is added */
649  }
650  else if((ota_pkt_idx + 1) > data_len) {
651  debugger[1]++;
652  /* last frame already sent */
653  ota_state_machine_g = OTA_DATAREQ;
654  if(ymodem_state_machine_g == YMODEM_CLOSE) {
655  while(1);
656  }
657  app_token = TOKEN_RELEASE;
658  return 0;
659  }
660  /* still a few byte to send, it is the last frame */
661  else if((ota_pkt_idx + 1) <= data_len) {
662  debugger[2]++;
663  ota_pkt_size = (data_len - ota_pkt_idx) + 2; /* +2 because the sequence number is added */
664  if(ymodem_state_machine_g == YMODEM_CLOSE)
665  last_frame = 1;
666  }
667  else {
668  while(1); /* not possible ? */
669  }
670 
671  /* copy the buffer */
672  for(uint8_t i = 0; i < ota_pkt_size; i++) {
673  image_tmp[4+i] = image[ota_pkt_idx%1024 + i];
674  }
675 
676  image_tmp[0] = HEADER_SENDATA;
677  image_tmp[1] = ota_pkt_size;
678  image_tmp[2] = (uint8_t)(seq_num>>8);
679  image_tmp[3] = (uint8_t)seq_num;
680 
681  /* usare dei puntatori per muoversi nell'array dell'image */
682  debugger[3] = ota_state_machine_g;
683  ota_state_machine_g = OTA_ONGOING;
684  RADIO_SetBackToBackTime(150);
685  ret = HAL_RADIO_SendPacketWithAck(OTA_CHANNEL, TX_WAKEUP_TIME, (uint8_t*)image_tmp, rx_ota_buffer, RX_TIMEOUT_ACK, MAX_LL_PACKET_LENGTH, OTA_SendDataCallback);
686  if(ret != SUCCESS_0) {
687  PRINTF("ERROR %d\r\n",ret);
688  }
689  }
690 
691  else if(ota_state_machine_g == OTA_COMPLETE) {
692  PRINTF("OTA_COMPLETE\r\n");
693 
694  PRINTF("image %d\r\n",image_size);
695  ota_state_machine_g = OTA_CONNECTION;
696  last_frame = 0;
697  seq_num = 0;
698  app_token = TOKEN_RELEASE;
699  }
700 
701  if (bootloadingCompleted) {
702  bootloadingCompleted = 0;
703  }
704 
705  return (bootloadingCompleted_end);
706 }

715 int32_t OTA_ymodem_tick(void)
716 {
717  int32_t ret = 0;
718  static volatile uint32_t packet = 0;
719  uint32_t ymodem_data_len = 0;
720  volatile uint32_t tmp_size;
721  uint8_t tmp_buff[1024+8];
722 
723  if(app_token == TOKEN_TAKEN_OTA)
724  return 0;
725  app_token = TOKEN_TAKEN_YMODEM;
726 
727  /* Get file name and size */
728  if(ymodem_state_machine_g == YMODEM_SIZE) {
729  ret = Ymodem_Receive ((uint8_t *)tmp_buff, 0, (uint32_t *) &image_size, packet++);
730  if (ret != YMODEM_CONTINUE) {
731  ymodem_state_machine_g = YMODEM_ABORT;
732  }
733  else {
734  ymodem_state_machine_g = YMODEM_LOAD;
735  app_token = TOKEN_RELEASE;
736  }
737  }
738 
739  else if(ymodem_state_machine_g == YMODEM_LOAD) {
740 
741  /* Answer to the previous Receive */
742 
743  Ymodem_SendAck();
744 
745  ret = Ymodem_Receive (image, 0, (uint32_t *) &ymodem_data_len, packet++);
746  if (ret != YMODEM_CONTINUE) {
747  ymodem_state_machine_g = YMODEM_ABORT;
748  return ret;
749  }
750  else {
751  app_token = TOKEN_RELEASE;
752  }
753 
754  if((ymodem_data_len+data_len) > image_size) {
755  ymodem_state_machine_g = YMODEM_CLOSE;
756 
757  /* Answer to the previous Receive */
758  Ymodem_SendAck();
759  data_len = image_size;
760  }
761  else {
762  data_len += ymodem_data_len;
763  ymodem_state_machine_g = YMODEM_WAIT;
764  }
765  }
766 
767  else if(ymodem_state_machine_g == YMODEM_WAIT) {
768  /* Empty state waiting for OTA action. Still data with ymodem. */
769  ymodem_state_machine_g = YMODEM_LOAD;
770  }
771 
772  else if(ymodem_state_machine_g == YMODEM_CLOSE) {
773  ymodem_state_machine_g = YMODEM_COMPLETE;
774 
775  /* Close Ymodem session */
776  ret = Ymodem_Receive (tmp_buff, 0, (uint32_t *) &tmp_size, packet++);
777  if (ret != YMODEM_DONE) {
778  Ymodem_Abort();
779  }
780  Ymodem_SendAck();
781  }
782 
783  else if(ymodem_state_machine_g == YMODEM_COMPLETE) {
784 
785  }
786 
787  else if(ymodem_state_machine_g == YMODEM_ABORT) {
788  Ymodem_Abort();
789  /* Reset all the states and variables or reset the system */
790  }
791  return 0;
792 }
