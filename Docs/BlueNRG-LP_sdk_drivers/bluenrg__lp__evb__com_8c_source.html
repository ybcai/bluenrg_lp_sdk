224 uint8_t BSP_COM_Read(uint8_t *pData)
225 {
226  uint8_t tmp = 0;
227 
228  /* If the user callback is not NULL */
229  if(BSP_COM_RxDataCb.RxDataUserCb != NULL) {
230  tmp = Read_Buffer_Pop(pData);
231 
232 #ifdef __CC_ARM
233  if(*(int *)pData == EOF)
234  return 0;
235 #endif
236  }
237  /* If the user callback is NULL, then the management of the reception is done manually */
238  else {
239  if(LL_USART_IsActiveFlag_RXNE(BSP_UART) == 1) {
240  *pData = LL_USART_ReceiveData8(BSP_UART);
241  tmp = 1;
242  }
243  }
244 
245  return tmp; /* 1: character has been read, 0: no character */
246 }
